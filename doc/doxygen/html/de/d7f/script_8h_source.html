370  explicit CScriptNum(const std::vector<unsigned char>& vch, bool fRequireMinimal)
371  {
372  if (vch.size() > nMaxNumSize) {
373  throw scriptnum_error("script number overflow");
374  }
375  if (fRequireMinimal && vch.size() > 0) {
376  // Check that the number is encoded with the minimum possible
377  // number of bytes.
378  //
379  // If the most-significant-byte - excluding the sign bit - is zero
380  // then we're not minimal. Note how this test also rejects the
381  // negative-zero encoding, 0x80.
382  if ((vch.back() & 0x7f) == 0) {
383  // One exception: if there's more than one byte and the most
384  // significant bit of the second-most-significant-byte is set
385  // it would conflict with the sign bit. An example of this case
386  // is +-255, which encode to 0xff00 and 0xff80 respectively.
387  // (big-endian).
388  if (vch.size() <= 1 || (vch[vch.size() - 2] & 0x80) == 0) {
389  throw scriptnum_error("non-minimally encoded script number");
390  }
391  }
392  }
393  m_value = set_vch(vch);
394  }

460  static std::vector<unsigned char> serialize(const int64_t& value)
461  {
462  if(value == 0)
463  return std::vector<unsigned char>();
464 
465  std::vector<unsigned char> result;
466  const bool neg = value < 0;
467  uint64_t absvalue = neg ? -value : value;
468 
469  while(absvalue)
470  {
471  result.push_back(absvalue & 0xff);
472  absvalue >>= 8;
473  }
474 
475 // - If the most significant byte is >= 0x80 and the value is positive, push a
476 // new zero-byte to make the significant byte < 0x80 again.
477 
478 // - If the most significant byte is >= 0x80 and the value is negative, push a
479 // new 0x80 byte that will be popped off when converting to an integral.
480 
481 // - If the most significant byte is < 0x80 and the value is negative, add
482 // 0x80 to it, since it will be subtracted and interpreted as a negative when
483 // converting to an integral.
484 
485  if (result.back() & 0x80)
486  result.push_back(neg ? 0x80 : 0);
487  else if (neg)
488  result.back() |= 0x80;
489 
490  return result;
491  }

495 private:
496  static int64_t set_vch(const std::vector<unsigned char>& vch)
497  {
498  if (vch.empty())
499  return 0;
500 
501  int64_t result = 0;
502  for (size_t i = 0; i != vch.size(); ++i)
503  result |= static_cast<int64_t>(vch[i]) << 8*i;
504 
505  // If the input vector's most significant byte is 0x80, remove it from
506  // the result's msb and return a negative.
507  if (vch.back() & 0x80)
508  return -((int64_t)(result & ~(0x80ULL << (8 * (vch.size() - 1)))));
509 
510  return result;
511  }

666  bool GetOp2(const_iterator& pc, opcodetype& opcodeRet, std::vector<unsigned char>* pvchRet) const
667  {
668  opcodeRet = OP_INVALIDOPCODE;
669  if (pvchRet)
670  pvchRet->clear();
671  if (pc >= end())
672  return false;
673 
674  // Read instruction
675  if (end() - pc < 1)
676  return false;
677  unsigned int opcode = *pc++;
678 
679  // Immediate operand
680  if (opcode <= OP_PUSHDATA4)
681  {
682  unsigned int nSize;
683  if (opcode < OP_PUSHDATA1)
684  {
685  nSize = opcode;
686  }
687  else if (opcode == OP_PUSHDATA1)
688  {
689  if (end() - pc < 1)
690  return false;
691  nSize = *pc++;
692  }
693  else if (opcode == OP_PUSHDATA2)
694  {
695  if (end() - pc < 2)
696  return false;
697  nSize = 0;
698  memcpy(&nSize, &pc[0], 2);
699  pc += 2;
700  }
701  else if (opcode == OP_PUSHDATA4)
702  {
703  if (end() - pc < 4)
704  return false;
705  memcpy(&nSize, &pc[0], 4);
706  pc += 4;
707  }
708  if (end() - pc < 0 || (unsigned int)(end() - pc) < nSize)
709  return false;
710  if (pvchRet)
711  pvchRet->assign(pc, pc + nSize);
712  pc += nSize;
713  }
714 
715  opcodeRet = (opcodetype)opcode;
716  return true;
717  }
