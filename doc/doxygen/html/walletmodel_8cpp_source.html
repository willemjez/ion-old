129 void WalletModel::pollBalanceChanged()
130 {
131  // Get required locks upfront. This avoids the GUI from getting stuck on
132  // periodical polls if the core is holding the locks for a longer time -
133  // for example, during a wallet rescan.
134  TRY_LOCK(cs_main, lockMain);
135  if(!lockMain)
136  return;
137  TRY_LOCK(wallet->cs_wallet, lockWallet);
138  if(!lockWallet)
139  return;
140 
141  if(fForceCheckBalanceChanged || nBestHeight != cachedNumBlocks || nStashedsendRounds != cachedStashedsendRounds || cachedTxLocks != nCompleteTXLocks)
142  {
143  fForceCheckBalanceChanged = false;
144 
145  // Balance and number of transactions might have changed
146  cachedNumBlocks = nBestHeight;
147  cachedStashedsendRounds = nStashedsendRounds;
148 
149  checkBalanceChanged();
150  if(transactionTableModel)
151  transactionTableModel->updateConfirmations();
152  }
153 }

178  cachedTxLocks != nCompleteTXLocks || cachedWatchOnlyBalance != newWatchOnlyBalance || cachedWatchOnlyStake != newWatchOnlyStake || cachedWatchUnconfBalance != newWatchUnconfBalance || cachedWatchImmatureBalance != newWatchImmatureBalance)

226 WalletModel::SendCoinsReturn WalletModel::prepareTransaction(WalletModelTransaction &transaction, const CCoinControl *coinControl)
227 {
228  CAmount total = 0;
229  QList<SendCoinsRecipient> recipients = transaction.getRecipients();
230  std::vector<std::pair<CScript, CAmount> > vecSend;
231 
232  if(recipients.empty())
233  {
234  return OK;
235  }
236 
237  if(isAnonymizeOnlyUnlocked())
238  {
239  return AnonymizeOnlyUnlocked;
240  }
241 
242  QSet<QString> setAddress; // Used to detect duplicates
243  int nAddresses = 0;
244 
245  // Pre-check input data for validity
246  foreach(const SendCoinsRecipient &rcp, recipients)
247  {
248  if(!validateAddress(rcp.address))
249  {
250  return InvalidAddress;
251  }
252  if(rcp.amount <= 0)
253  {
254  return InvalidAmount;
255  }
256  setAddress.insert(rcp.address);
257  ++nAddresses;
258 
259  CScript scriptPubKey = GetScriptForDestination(CIonAddress(rcp.address.toStdString()).Get());
260  vecSend.push_back(std::pair<CScript, CAmount>(scriptPubKey, rcp.amount));
261 
262  total += rcp.amount;
263  }
264  if(setAddress.size() != nAddresses)
265  {
266  return DuplicateAddress;
267  }
268 
269  CAmount nBalance = getBalance(coinControl);
270 
271  if(total > nBalance)
272  {
273  return AmountExceedsBalance;
274  }
275 
276  {
277  LOCK2(cs_main, wallet->cs_wallet);
278 
279  transaction.newPossibleKeyChange(wallet);
280  CAmount nFeeRequired = 0;
281  std::string strFailReason;
282 
283  CWalletTx *newTx = transaction.getTransaction();
284  CReserveKey *keyChange = transaction.getPossibleKeyChange();
285 
286  if(recipients[0].useInstantX && total > GetSporkValue(SPORK_5_MAX_VALUE)*COIN){
287  return IXTransactionCreationFailed;
288  }
289 
290  int nChangePos;
291  bool fCreated = wallet->CreateTransaction(vecSend, *newTx, *keyChange, nFeeRequired, nChangePos, strFailReason, coinControl, recipients[0].inputType, recipients[0].useInstantX);
292  transaction.setTransactionFee(nFeeRequired);
293 
294  if(recipients[0].useInstantX && newTx->GetValueOut() > GetSporkValue(SPORK_5_MAX_VALUE)*COIN){
295  return IXTransactionCreationFailed;
296  }
297 
298  if(!fCreated)
299  {
300  if((total + nFeeRequired) > nBalance)
301  {
302  return SendCoinsReturn(AmountWithFeeExceedsBalance);
303  }
304  emit message(tr("Send Coins"), QString::fromStdString(strFailReason), false,
305  CClientUIInterface::MSG_ERROR);
306  return PrepareTransactionFailed;
307  }
308  // reject insane fee
309  unsigned int insanefee = (transaction.getTransactionSize() == 0 ? nAddresses : transaction.getTransactionSize());
310  insanefee = ((MIN_RELAY_TX_FEE * insanefee) * 10000);
311  if (nFeeRequired > insanefee){
312  LogPrintf("nFeeRequired: %d -- InsaneFee: %d\n", nFeeRequired, insanefee);
313  return InsaneFee;
314  }
315  }
316 
317  return SendCoinsReturn(OK);
318 }

320 WalletModel::SendCoinsReturn WalletModel::sendCoins(WalletModelTransaction &transaction, const CCoinControl *coinControl)
321 {
322  QByteArray transaction_array; /* store serialized transaction */
323  CAmount total = 0;
324  QSet<QString> setAddress;
325  QList<SendCoinsRecipient> recipients = transaction.getRecipients();
326  std::vector<std::pair<CScript, CAmount> > vecSend;
327 
328  if(isAnonymizeOnlyUnlocked())
329  {
330  return AnonymizeOnlyUnlocked;
331  }
332 
333  std::map<int, std::string> mapStealthNarr;
334 
335  {
336  LOCK2(cs_main, wallet->cs_wallet);
337 
338  CWalletTx *newTx = transaction.getTransaction();
339  CWalletTx wtx;
340 
341 
342  // Sendmany
343  std::vector<std::pair<CScript, int64_t> > vecSend;
344  foreach(const SendCoinsRecipient &rcp, recipients)
345  {
346  std::string sAddr = rcp.address.toStdString();
347 
348  if (rcp.typeInd == AddressTableModel::AT_Stealth)
349  {
350  CStealthAddress sxAddr;
351  if (sxAddr.SetEncoded(sAddr))
352  {
353  ec_secret ephem_secret;
354  ec_secret secretShared;
355  ec_point pkSendTo;
356  ec_point ephem_pubkey;
357 
358 
359  if (GenerateRandomSecret(ephem_secret) != 0)
360  {
361  printf("GenerateRandomSecret failed.\n");
362  return Aborted;
363  };
364 
365  if (StealthSecret(ephem_secret, sxAddr.scan_pubkey, sxAddr.spend_pubkey, secretShared, pkSendTo) != 0)
366  {
367  printf("Could not generate receiving public key.\n");
368  return Aborted;
369  };
370 
371  CPubKey cpkTo(pkSendTo);
372  if (!cpkTo.IsValid())
373  {
374  printf("Invalid public key generated.\n");
375  return Aborted;
376  };
377 
378  CKeyID ckidTo = cpkTo.GetID();
379 
380  CIonAddress addrTo(ckidTo);
381 
382  if (SecretToPublicKey(ephem_secret, ephem_pubkey) != 0)
383  {
384  printf("Could not generate ephem public key.\n");
385  return Aborted;
386  };
387 
388  if (fDebug)
389  {
390  printf("Stealth send to generated pubkey %" PRIszu": %s\n", pkSendTo.size(), HexStr(pkSendTo).c_str());
391  printf("hash %s\n", addrTo.ToString().c_str());
392  printf("ephem_pubkey %" PRIszu": %s\n", ephem_pubkey.size(), HexStr(ephem_pubkey).c_str());
393  };
394 
395  CScript scriptPubKey;
396  scriptPubKey.SetDestination(addrTo.Get());
397 
398  vecSend.push_back(make_pair(scriptPubKey, rcp.amount));
399 
400  CScript scriptP = CScript() << OP_RETURN << ephem_pubkey;
401 
402  if (rcp.narration.length() > 0)
403  {
404  std::string sNarr = rcp.narration.toStdString();
405 
406  if (sNarr.length() > 24)
407  {
408  printf("Narration is too long.\n");
409  return NarrationTooLong;
410  };
411 
412  std::vector<unsigned char> vchNarr;
413 
414  SecMsgCrypter crypter;
415  crypter.SetKey(&secretShared.e[0], &ephem_pubkey[0]);
416 
417  if (!crypter.Encrypt((uint8_t*)&sNarr[0], sNarr.length(), vchNarr))
418  {
419  printf("Narration encryption failed.\n");
420  return Aborted;
421  };
422 
423  if (vchNarr.size() > 48)
424  {
425  printf("Encrypted narration is too long.\n");
426  return Aborted;
427  };
428 
429  if (vchNarr.size() > 0)
430  scriptP = scriptP << OP_RETURN << vchNarr;
431 
432  int pos = vecSend.size()-1;
433  mapStealthNarr[pos] = sNarr;
434  };
435 
436  vecSend.push_back(make_pair(scriptP, 0));
437 
438  continue;
439  } else {
440  printf("Couldn't parse stealth address!\n");
441  return Aborted;
442  } // else drop through to normal
443  }
444 
445  CScript scriptPubKey;
446  scriptPubKey.SetDestination(CIonAddress(sAddr).Get());
447  vecSend.push_back(make_pair(scriptPubKey, rcp.amount));
448 
449  if (rcp.narration.length() > 0)
450  {
451  std::string sNarr = rcp.narration.toStdString();
452 
453  if (sNarr.length() > 24)
454  {
455  printf("Narration is too long.\n");
456  return NarrationTooLong;
457  };
458 
459  std::vector<uint8_t> vNarr(sNarr.c_str(), sNarr.c_str() + sNarr.length());
460  std::vector<uint8_t> vNDesc;
461 
462  vNDesc.resize(2);
463  vNDesc[0] = 'n';
464  vNDesc[1] = 'p';
465 
466  CScript scriptN = CScript() << OP_RETURN << vNDesc << OP_RETURN << vNarr;
467 
468  vecSend.push_back(make_pair(scriptN, 1));
469  }
470  }
471 
472  CReserveKey keyChange(wallet);
473  CAmount nFeeRequired = 0;
474  int nChangePos = -1;
475  std::string strFailReason;
476 
477  bool fCreated = wallet->CreateTransaction(vecSend, wtx, keyChange, nFeeRequired, nChangePos, strFailReason, coinControl, recipients[0].inputType, recipients[0].useInstantX);
478  transaction.setTransactionFee(nFeeRequired);
479 
480  std::map<int, std::string>::iterator it;
481  for (it = mapStealthNarr.begin(); it != mapStealthNarr.end(); ++it)
482  {
483  int pos = it->first;
484  if (nChangePos > -1 && it->first >= nChangePos)
485  pos++;
486 
487  char key[64];
488  if (snprintf(key, sizeof(key), "n_%u", pos) < 1)
489  {
490  printf("CreateStealthTransaction(): Error creating narration key.");
491  continue;
492  };
493  wtx.mapValue[key] = it->second;
494  };
495 
496  CAmount nBalance = getBalance(coinControl);
497 
498  if(!fCreated)
499  {
500  if(total > nBalance)
501  {
502  return AmountExceedsBalance;
503  }
504  if((total + nFeeRequired) > nBalance) // FIXME: could cause collisions in the future
505  {
506  return SendCoinsReturn(AmountWithFeeExceedsBalance);
507  }
508  return TransactionCreationFailed;
509  }
510  if(!uiInterface.ThreadSafeAskFee(nFeeRequired, tr("Sending...").toStdString()))
511  {
512  return Aborted;
513  }
514  if(!wallet->CommitTransaction(wtx, keyChange, (recipients[0].useInstantX) ? "txlreq" : "tx"))
515  {
516  return TransactionCommitFailed;
517  }
518  }
519 
520  // Add addresses / update labels that we've sent to to the address book
521  foreach(const SendCoinsRecipient &rcp, recipients)
522  {
523  std::string strAddress = rcp.address.toStdString();
524  CTxDestination dest = CIonAddress(strAddress).Get();
525  std::string strLabel = rcp.label.toStdString();
526  {
527  LOCK(wallet->cs_wallet);
528 
529  if (rcp.typeInd == AddressTableModel::AT_Stealth)
530  {
531  wallet->UpdateStealthAddress(strAddress, strLabel, true);
532  } else
533  {
534  std::map<CTxDestination, std::string>::iterator mi = wallet->mapAddressBook.find(dest);
535 
536  // Check if we have a new address or an updated label
537  if (mi == wallet->mapAddressBook.end() || mi->second != strLabel)
538  {
539  wallet->SetAddressBookName(dest, strLabel);
540  }
541  }
542  }
543  emit coinsSent(wallet, rcp, transaction_array);
544  }
545  checkBalanceChanged(); // update balance immediately, otherwise there could be a short noticeable delay until pollBalanceChanged hits
546 
547  return SendCoinsReturn(OK);
548 }

691 static void ShowProgress(WalletModel *walletmodel, const std::string &title, int nProgress)
692 {
693  // emits signal "showProgress"
694  QMetaObject::invokeMethod(walletmodel, "showProgress", Qt::QueuedConnection,
695  Q_ARG(QString, QString::fromStdString(title)),
696  Q_ARG(int, nProgress));
697  if (nProgress == 0)
698  fQueueNotifications = true;
699 
700  if (nProgress == 100)
701  {
702  fQueueNotifications = false;
703  BOOST_FOREACH(const PAIRTYPE(uint256, ChangeType)& notification, vQueueNotifications)
704  NotifyTransactionChanged(walletmodel, NULL, notification.first, notification.second);
705  if (vQueueNotifications.size() > 10) // prevent balloon spam, show maximum 10 balloons
706  QMetaObject::invokeMethod(walletmodel, "setProcessingQueuedTransactions", Qt::QueuedConnection, Q_ARG(bool, true));
