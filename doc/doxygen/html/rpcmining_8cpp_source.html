155 Value checkkernel(const Array& params, bool fHelp)
156 {
157  if (fHelp || params.size() < 1 || params.size() > 2)
158  throw runtime_error(
159  "checkkernel [{\"txid\":txid,\"vout\":n},...] [createblocktemplate=false]\n"
160  "Check if one of given inputs is a kernel input at the moment.\n"
161  );
162 
163  RPCTypeCheck(params, list_of(array_type)(bool_type));
164 
165  Array inputs = params[0].get_array();
166  bool fCreateBlockTemplate = params.size() > 1 ? params[1].get_bool() : false;
167 
168  if (vNodes.empty())
169  throw JSONRPCError(-9, "Ion is not connected!");
170 
171  if (IsInitialBlockDownload())
172  throw JSONRPCError(-10, "Ion is downloading blocks...");
173 
174  COutPoint kernel;
175  CBlockIndex* pindexPrev = pindexBest;
176  unsigned int nBits = GetNextTargetRequired(pindexPrev, true);
177  int64_t nTime = GetAdjustedTime();
178  nTime &= ~STAKE_TIMESTAMP_MASK;
179 
180  BOOST_FOREACH(Value& input, inputs)
181  {
182  const Object& o = input.get_obj();
183 
184  const Value& txid_v = find_value(o, "txid");
185  if (txid_v.type() != str_type)
186  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, missing txid key");
187  string txid = txid_v.get_str();
188  if (!IsHex(txid))
189  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, expected hex txid");
190 
191  const Value& vout_v = find_value(o, "vout");
192  if (vout_v.type() != int_type)
193  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, missing vout key");
194  int nOutput = vout_v.get_int();
195  if (nOutput < 0)
196  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, vout must be positive");
197 
198  COutPoint cInput(uint256(txid), nOutput);
199  if (CheckKernel(pindexPrev, nBits, nTime, cInput))
200  {
201  kernel = cInput;
202  break;
203  }
204  }
205 
206  Object result;
207  result.push_back(Pair("found", !kernel.IsNull()));
208 
209  if (kernel.IsNull())
210  return result;
211 
212  Object oKernel;
213  oKernel.push_back(Pair("txid", kernel.hash.GetHex()));
214  oKernel.push_back(Pair("vout", (int64_t)kernel.n));
215  oKernel.push_back(Pair("time", nTime));
216  result.push_back(Pair("kernel", oKernel));
217 
218  if (!fCreateBlockTemplate)
219  return result;
220 
221  CAmount nFees;
222  auto_ptr<CBlock> pblock(CreateNewBlock(*pMiningKey, true, &nFees));
223 
224  pblock->nTime = pblock->vtx[0].nTime = nTime;
225 
226  CDataStream ss(SER_DISK, PROTOCOL_VERSION);
227  ss << *pblock;
228 
229  result.push_back(Pair("blocktemplate", HexStr(ss.begin(), ss.end())));
230  result.push_back(Pair("blocktemplatefees", nFees));
231 
232  CPubKey pubkey;
233  if (!pMiningKey->GetReservedKey(pubkey))
234  throw JSONRPCError(RPC_MISC_ERROR, "GetReservedKey failed");
235 
236  result.push_back(Pair("blocktemplatesignkey", HexStr(pubkey)));
237 
238  return result;
239 }

241 Value getworkex(const Array& params, bool fHelp)
242 {
243  if (fHelp || params.size() > 2)

246  "If [data, coinbase] is not specified, returns extended work data.\n"
247  );
248

253  throw JSONRPCError(-10, "Ion is downloading blocks...");

259  static mapNewBlock_t mapNewBlock;
260  static vector<CBlock*> vNewBlock;
261 
262  if (params.size() == 0)
263  {
264  // Update block
265  static unsigned int nTransactionsUpdatedLast;
266  static CBlockIndex* pindexPrev;
267  static int64_t nStart;
268  static CBlock* pblock;
269  if (pindexPrev != pindexBest ||
270  (mempool.GetTransactionsUpdated() != nTransactionsUpdatedLast && GetTime() - nStart > 60))
271  {
272  if (pindexPrev != pindexBest)
273  {
274  // Deallocate old blocks since they're obsolete now
275  mapNewBlock.clear();
276  BOOST_FOREACH(CBlock* pblock, vNewBlock)
277  delete pblock;
278  vNewBlock.clear();
279  }
280  nTransactionsUpdatedLast = mempool.GetTransactionsUpdated();
281  pindexPrev = pindexBest;
282  nStart = GetTime();
283 
284  // Create new block
285  pblock = CreateNewBlock(*pMiningKey);
286  if (!pblock)
287  throw JSONRPCError(-7, "Out of memory");
288  vNewBlock.push_back(pblock);
289  }
290 
291  // Update nTime
292  pblock->nTime = max(pindexPrev->GetPastTimeLimit()+1, GetAdjustedTime());
293  pblock->nNonce = 0;
294 
295  // Update nExtraNonce
296  static unsigned int nExtraNonce = 0;
297  IncrementExtraNonce(pblock, pindexPrev, nExtraNonce);

299  // Save
300  mapNewBlock[pblock->hashMerkleRoot] = make_pair(pblock, pblock->vtx[0].vin[0].scriptSig);
301 
302  // Prebuild hash buffers
303  char pmidstate[32];
304  char pdata[128];
305  char phash1[64];
306  FormatHashBuffers(pblock, pmidstate, pdata, phash1);
307 
308  uint256 hashTarget = uint256().SetCompact(pblock->nBits);
309 
310  CTransaction coinbaseTx = pblock->vtx[0];
311  std::vector<uint256> merkle = pblock->GetMerkleBranch(0);
312 
313  Object result;
314  result.push_back(Pair("data", HexStr(BEGIN(pdata), END(pdata))));
315  result.push_back(Pair("target", HexStr(BEGIN(hashTarget), END(hashTarget))));
316 
317  CDataStream ssTx(SER_NETWORK, PROTOCOL_VERSION);
318  ssTx << coinbaseTx;
319  result.push_back(Pair("coinbase", HexStr(ssTx.begin(), ssTx.end())));
320 
321  Array merkle_arr;
322 
323  BOOST_FOREACH(uint256 merkleh, merkle) {
324  merkle_arr.push_back(HexStr(BEGIN(merkleh), END(merkleh)));
325  }
326 
327  result.push_back(Pair("merkle", merkle_arr));
328 
329 
330  return result;
331  }
332  else
333  {
334  // Parse parameters
335  vector<unsigned char> vchData = ParseHex(params[0].get_str());
336  vector<unsigned char> coinbase;
337 
338  if(params.size() == 2)
339  coinbase = ParseHex(params[1].get_str());
340 
341  if (vchData.size() != 128)
342  throw JSONRPCError(-8, "Invalid parameter");
343 
344  CBlock* pdata = (CBlock*)&vchData[0];
345 
346  // Byte reverse
347  for (int i = 0; i < 128/4; i++)
348  ((unsigned int*)pdata)[i] = ByteReverse(((unsigned int*)pdata)[i]);
349 
350  // Get saved block
351  if (!mapNewBlock.count(pdata->hashMerkleRoot))
352  return false;
353  CBlock* pblock = mapNewBlock[pdata->hashMerkleRoot].first;
354 
355  pblock->nTime = pdata->nTime;
356  pblock->nNonce = pdata->nNonce;
357 
358  if(coinbase.size() == 0)
359  pblock->vtx[0].vin[0].scriptSig = mapNewBlock[pdata->hashMerkleRoot].second;
360  else
361  CDataStream(coinbase, SER_NETWORK, PROTOCOL_VERSION) >> pblock->vtx[0]; // FIXME - HACK!
362 
363  pblock->hashMerkleRoot = pblock->BuildMerkleTree();
364 
365  assert(pwalletMain != NULL);
366  return CheckWork(pblock, *pwalletMain, *pMiningKey);
367  }
