355 bool IsDERSignature(const valtype &vchSig, bool haveHashType) {
356  // See https://bitcointalk.org/index.php?topic=8392.msg127623#msg127623
357  // A canonical signature exists of: <30> <total len> <02> <len R> <R> <02> <len S> <S> <hashtype>
358  // Where R and S are not negative (their first byte has its highest bit not set), and not
359  // excessively padded (do not start with a 0 byte, unless an otherwise negative number follows,
360  // in which case a single 0 byte is necessary and even required).
361  if (vchSig.size() < 9)
362  return error("Non-canonical signature: too short");
363  if (vchSig.size() > 73)
364  return error("Non-canonical signature: too long");
365  if (vchSig[0] != 0x30)
366  return error("Non-canonical signature: wrong type");
367  if (vchSig[1] != vchSig.size() - (haveHashType ? 3 : 2))
368  return error("Non-canonical signature: wrong length marker");
369  unsigned int nLenR = vchSig[3];
370  if (5 + nLenR >= vchSig.size())
371  return error("Non-canonical signature: S length misplaced");
372  unsigned int nLenS = vchSig[5+nLenR];
373  if ((unsigned long)(nLenR + nLenS + (haveHashType ? 7 : 6)) != vchSig.size())
374  return error("Non-canonical signature: R+S length mismatch");
375 
376  const unsigned char *R = &vchSig[4];
377  if (R[-2] != 0x02)
378  return error("Non-canonical signature: R value type mismatch");
379  if (nLenR == 0)
380  return error("Non-canonical signature: R length is zero");
381  if (R[0] & 0x80)
382  return error("Non-canonical signature: R value negative");
383  if (nLenR > 1 && (R[0] == 0x00) && !(R[1] & 0x80))
384  return error("Non-canonical signature: R value excessively padded");
385 
386  const unsigned char *S = &vchSig[6+nLenR];
387  if (S[-2] != 0x02)
388  return error("Non-canonical signature: S value type mismatch");
389  if (nLenS == 0)
390  return error("Non-canonical signature: S length is zero");
391  if (S[0] & 0x80)
392  return error("Non-canonical signature: S value negative");
393  if (nLenS > 1 && (S[0] == 0x00) && !(S[1] & 0x80))
394  return error("Non-canonical signature: S value excessively padded");
395 
396  return true;
397 }

434 bool EvalScript(vector<vector<unsigned char> >& stack, const CScript& script, const CTransaction& txTo, unsigned int nIn, unsigned int flags, int nHashType)
435 {
436  static const CScriptNum bnZero(0);
437  static const CScriptNum bnOne(1);
438  static const CScriptNum bnFalse(0);
439  static const CScriptNum bnTrue(1);
440  static const valtype vchFalse(0);
441  static const valtype vchZero(0);
442  static const valtype vchTrue(1, 1);
443 
444  CScript::const_iterator pc = script.begin();
445  CScript::const_iterator pend = script.end();
446  CScript::const_iterator pbegincodehash = script.begin();
447  opcodetype opcode;
448  valtype vchPushValue;
449  vector<bool> vfExec;
450  vector<valtype> altstack;
451  if (script.size() > 10000)
452  return false;
453  int nOpCount = 0;
454  bool fRequireMinimal = (flags & SCRIPT_VERIFY_MINIMALDATA) != 0;
455 
456  try
457  {
458  while (pc < pend)
459  {
460  bool fExec = !count(vfExec.begin(), vfExec.end(), false);
461 
462  //
463  // Read instruction
464  //
465  if (!script.GetOp(pc, opcode, vchPushValue))
466  return false;
467  if (vchPushValue.size() > MAX_SCRIPT_ELEMENT_SIZE)
468  return false;
469  if (opcode > OP_16 && ++nOpCount > 201)
470  return false;
471 
472  if (opcode == OP_CAT ||
473  opcode == OP_SUBSTR ||
474  opcode == OP_LEFT ||
475  opcode == OP_RIGHT ||
476  opcode == OP_INVERT ||
477  opcode == OP_AND ||
478  opcode == OP_OR ||
479  opcode == OP_XOR ||
480  opcode == OP_2MUL ||
481  opcode == OP_2DIV ||
482  opcode == OP_MUL ||
483  opcode == OP_DIV ||
484  opcode == OP_MOD ||
485  opcode == OP_LSHIFT ||
486  opcode == OP_RSHIFT)
487  return false;
488 
489  if (fExec && 0 <= opcode && opcode <= OP_PUSHDATA4)
490  stack.push_back(vchPushValue);
491  else if (fExec || (OP_IF <= opcode && opcode <= OP_ENDIF))
492  switch (opcode)
493  {
494  //
495  // Push value
496  //
497  case OP_1NEGATE:
498  case OP_1:
499  case OP_2:
500  case OP_3:
501  case OP_4:
502  case OP_5:
503  case OP_6:
504  case OP_7:
505  case OP_8:
506  case OP_9:
507  case OP_10:
508  case OP_11:
509  case OP_12:
510  case OP_13:
511  case OP_14:
512  case OP_15:
513  case OP_16:
514  {
515  // ( -- value)
516  CScriptNum bn((int)opcode - (int)(OP_1 - 1));
517  stack.push_back(bn.getvch());
518  }
519  break;
520 
521 
522  //
523  // Control
524  //
525  case OP_NOP:
526  break;
527 
528  case OP_NOP1: case OP_NOP2: case OP_NOP3: case OP_NOP4: case OP_NOP5:
529  case OP_NOP6: case OP_NOP7: case OP_NOP8: case OP_NOP9: case OP_NOP10:
530  {
531  if (flags & SCRIPT_VERIFY_DISCOURAGE_UPGRADABLE_NOPS)
532  return false;
533  }
534  break;
535 
536  case OP_IF:
537  case OP_NOTIF:
538  {
539  // <expression> if [statements] [else [statements]] endif
540  bool fValue = false;
541  if (fExec)
542  {
543  if (stack.size() < 1)
544  return false;
545  valtype& vch = stacktop(-1);
546  fValue = CastToBool(vch);
547  if (opcode == OP_NOTIF)
548  fValue = !fValue;
549  popstack(stack);
550  }
551  vfExec.push_back(fValue);
552  }
553  break;
554 
555  case OP_ELSE:
556  {
557  if (vfExec.empty())
558  return false;
559  vfExec.back() = !vfExec.back();
560  }
561  break;
562 
563  case OP_ENDIF:
564  {
565  if (vfExec.empty())
566  return false;
567  vfExec.pop_back();
568  }
569  break;
570 
571  case OP_VERIFY:
572  {
573  // (true -- ) or
574  // (false -- false) and return
575  if (stack.size() < 1)
576  return false;
577  bool fValue = CastToBool(stacktop(-1));
578  if (fValue)
579  popstack(stack);
580  else
581  return false;
582  }
583  break;
584 
585  case OP_RETURN:
586  {
587  return false;
588  }
589  break;
590 
591 
592  //
593  // Stack ops
594  //
595  case OP_TOALTSTACK:
596  {
597  if (stack.size() < 1)
598  return false;
599  altstack.push_back(stacktop(-1));
600  popstack(stack);
601  }
602  break;
603 
604  case OP_FROMALTSTACK:
605  {
606  if (altstack.size() < 1)
607  return false;
608  stack.push_back(altstacktop(-1));
609  popstack(altstack);
610  }
611  break;
612 
613  case OP_2DROP:
614  {
615  // (x1 x2 -- )
616  if (stack.size() < 2)
617  return false;
618  popstack(stack);
619  popstack(stack);
620  }
621  break;
622 
623  case OP_2DUP:
624  {
625  // (x1 x2 -- x1 x2 x1 x2)
626  if (stack.size() < 2)
627  return false;
628  valtype vch1 = stacktop(-2);
629  valtype vch2 = stacktop(-1);
630  stack.push_back(vch1);
631  stack.push_back(vch2);
632  }
633  break;
634 
635  case OP_3DUP:
636  {
637  // (x1 x2 x3 -- x1 x2 x3 x1 x2 x3)
638  if (stack.size() < 3)
639  return false;
640  valtype vch1 = stacktop(-3);
641  valtype vch2 = stacktop(-2);

1267  if (data.size() == 0) {
1268  // Could have used OP_0.
1269  return opcode == OP_0;

1271  // Could have used OP_1 .. OP_16.
