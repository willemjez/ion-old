18 bool NodeLessThan::operator()(const CNodeCombinedStats &left, const CNodeCombinedStats &right) const
19 {
20  const CNodeStats *pLeft = &(left.nodeStats);
21  const CNodeStats *pRight = &(right.nodeStats);
22 
23  if (order == Qt::DescendingOrder)
24  std::swap(pLeft, pRight);
25 
26  switch(column)
27  {
28  case PeerTableModel::Address:
29  return pLeft->addrName.compare(pRight->addrName) < 0;
30  case PeerTableModel::Subversion:
31  return pLeft->cleanSubVer.compare(pRight->cleanSubVer) < 0;
32  case PeerTableModel::Ping:
33  return pLeft->dPingTime < pRight->dPingTime;
34  }
35 
36  return false;

53  void refreshPeers()
54  {
55  {
56  TRY_LOCK(cs_vNodes, lockNodes);
57  if (!lockNodes)
58  {
59  // skip the refresh if we can't immediately get the lock
60  return;
61  }
62  cachedNodeStats.clear();
63 #if QT_VERSION >= 0x040700
64  cachedNodeStats.reserve(vNodes.size());
65 #endif
66  BOOST_FOREACH(CNode* pnode, vNodes)
67  {
68  CNodeCombinedStats stats;
69  stats.nodeStateStats.nMisbehavior = 0;
70  stats.fNodeStateStatsAvailable = false;
71  pnode->copyStats(stats.nodeStats);
72  cachedNodeStats.append(stats);
73  }
74  }
75 
76  // if we can, retrieve the CNodeStateStats for each node.
77  {
78  TRY_LOCK(cs_main, lockMain);
79  if (lockMain)
80  {
81  BOOST_FOREACH(CNodeCombinedStats &stats, cachedNodeStats)
82  {
83  stats.fNodeStateStatsAvailable = GetNodeStateStats(stats.nodeStats.nodeid, stats.nodeStateStats);
84  }
85  }
86  }
87 
88 
89  if (sortColumn >= 0) // sort cacheNodeStats (use stable sort to prevent rows jumping around unneceesarily)
90  qStableSort(cachedNodeStats.begin(), cachedNodeStats.end(), NodeLessThan(sortColumn, sortOrder));
91 
92  // build index map
93  mapNodeRows.clear();
94  int row = 0;
95  BOOST_FOREACH(CNodeCombinedStats &stats, cachedNodeStats)
96  {
97  mapNodeRows.insert(std::pair<NodeId, int>(stats.nodeStats.nodeid, row++));
98  }
99  }
100 
101  int size()
102  {
103  return cachedNodeStats.size();
104  }
105 
106  CNodeCombinedStats *index(int idx)
107  {
108  if(idx >= 0 && idx < cachedNodeStats.size()) {
109  return &cachedNodeStats[idx];
110  } else {
111  return 0;
112  }
113  }
