326 void OverviewPage::updateStashedsendProgress()
327 {
328  if(!stashedSendPool.IsBlockchainSynced() || ShutdownRequested()) return;
329 
330  if(!pwalletMain) return;
331 
332  QString strAmountAndRounds;
333  QString strAnonymizeIonAmount = IonUnits::formatHtmlWithUnit(nDisplayUnit, nAnonymizeIonAmount * COIN, false, IonUnits::separatorAlways);
334 
335  if(currentBalance == 0)
336  {
337  ui->stashedsendProgress->setValue(0);
338  ui->stashedsendProgress->setToolTip(tr("No inputs detected"));
339  // when balance is zero just show info from settings
340  strAnonymizeIonAmount = strAnonymizeIonAmount.remove(strAnonymizeIonAmount.indexOf("."), IonUnits::decimals(nDisplayUnit) + 1);
341  strAmountAndRounds = strAnonymizeIonAmount + " / " + tr("%n Rounds", "", nStashedsendRounds);
342

344  ui->labelAmountRounds->setText(strAmountAndRounds);
345  return;
346  }

348  CAmount nDenominatedConfirmedBalance;
349  CAmount nDenominatedUnconfirmedBalance;
350  CAmount nAnonymizableBalance;
351  CAmount nNormalizedAnonymizedBalance;
352  double nAverageAnonymizedRounds;
353 
354  {
355  TRY_LOCK(cs_main, lockMain);
356  if(!lockMain) return;
357 
358  nDenominatedConfirmedBalance = pwalletMain->GetDenominatedBalance();
359  nDenominatedUnconfirmedBalance = pwalletMain->GetDenominatedBalance(true);
360  nAnonymizableBalance = pwalletMain->GetAnonymizableBalance();
361  nNormalizedAnonymizedBalance = pwalletMain->GetNormalizedAnonymizedBalance();
362  nAverageAnonymizedRounds = pwalletMain->GetAverageAnonymizedRounds();
363  }
364 
365  //Get the anon threshold
366  CAmount nMaxToAnonymize = nAnonymizableBalance + currentAnonymizedBalance + nDenominatedUnconfirmedBalance;
367 
368  // If it's more than the anon threshold, limit to that.
369  if(nMaxToAnonymize > nAnonymizeIonAmount*COIN) nMaxToAnonymize = nAnonymizeIonAmount*COIN;
370 
371  if(nMaxToAnonymize == 0) return;
372 
373  if(nMaxToAnonymize >= nAnonymizeIonAmount * COIN) {
374  ui->labelAmountRounds->setToolTip(tr("Found enough compatible inputs to anonymize %1")
375  .arg(strAnonymizeIonAmount));
376  strAnonymizeIonAmount = strAnonymizeIonAmount.remove(strAnonymizeIonAmount.indexOf("."), IonUnits::decimals(nDisplayUnit) + 1);
377  strAmountAndRounds = strAnonymizeIonAmount + " / " + tr("%n Rounds", "", nStashedsendRounds);
378  } else {
379  QString strMaxToAnonymize = IonUnits::formatHtmlWithUnit(nDisplayUnit, nMaxToAnonymize, false, IonUnits::separatorAlways);
380  ui->labelAmountRounds->setToolTip(tr("Not enough compatible inputs to anonymize <span style='color:red;'>%1</span>,<br>"
381  "will anonymize <span style='color:red;'>%2</span> instead")
382  .arg(strAnonymizeIonAmount)
383  .arg(strMaxToAnonymize));
384  strMaxToAnonymize = strMaxToAnonymize.remove(strMaxToAnonymize.indexOf("."), IonUnits::decimals(nDisplayUnit) + 1);
385  strAmountAndRounds = "<span style='color:red;'>" +
386  QString(IonUnits::factor(nDisplayUnit) == 1 ? "" : "~") + strMaxToAnonymize +
387  " / " + tr("%n Rounds", "", nStashedsendRounds) + "</span>";
388  }
389  ui->labelAmountRounds->setText(strAmountAndRounds);
390 
391  // calculate parts of the progress, each of them shouldn't be higher than 1
392  // progress of denominating
393  float denomPart = 0;
394  // mixing progress of denominated balance
395  float anonNormPart = 0;
396  // completeness of full amount anonimization
397  float anonFullPart = 0;
398 
399  CAmount denominatedBalance = nDenominatedConfirmedBalance + nDenominatedUnconfirmedBalance;
400  denomPart = (float)denominatedBalance / nMaxToAnonymize;
401  denomPart = denomPart > 1 ? 1 : denomPart;
402  denomPart *= 100;
403 
404  anonNormPart = (float)nNormalizedAnonymizedBalance / nMaxToAnonymize;
405  anonNormPart = anonNormPart > 1 ? 1 : anonNormPart;
406  anonNormPart *= 100;
407 
408  anonFullPart = (float)currentAnonymizedBalance / nMaxToAnonymize;
409  anonFullPart = anonFullPart > 1 ? 1 : anonFullPart;
410  anonFullPart *= 100;
411 
412  // apply some weights to them ...
413  float denomWeight = 1;
414  float anonNormWeight = nStashedsendRounds;
415  float anonFullWeight = 2;
416  float fullWeight = denomWeight + anonNormWeight + anonFullWeight;
417  // ... and calculate the whole progress
418  float denomPartCalc = ceilf((denomPart * denomWeight / fullWeight) * 100) / 100;
419  float anonNormPartCalc = ceilf((anonNormPart * anonNormWeight / fullWeight) * 100) / 100;
420  float anonFullPartCalc = ceilf((anonFullPart * anonFullWeight / fullWeight) * 100) / 100;
421  float progress = denomPartCalc + anonNormPartCalc + anonFullPartCalc;
422  if(progress >= 100) progress = 100;
423
