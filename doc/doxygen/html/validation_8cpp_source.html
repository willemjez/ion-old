97  struct CBlockIndexWorkComparator
98  {
99  bool operator()(CBlockIndex *pa, CBlockIndex *pb) const {
100  // First sort by most total work, ...
101  if (pa->nChainWork > pb->nChainWork) return false;
102  if (pa->nChainWork < pb->nChainWork) return true;

104  // ... then by earliest time received, ...
105  if (pa->nSequenceId < pb->nSequenceId) return false;
106  if (pa->nSequenceId > pb->nSequenceId) return true;

108  // Use pointer address as tie breaker (should only happen with blocks
109  // loaded from disk, as those all have id 0).
110  if (pa < pb) return false;
111  if (pa > pb) return true;
112 
113  // Identical blocks.
114  return false;
115  }

194 bool CheckFinalTx(const CTransaction &tx, int flags)
195 {
196  AssertLockHeld(cs_main);
197 
198  // By convention a negative value for flags indicates that the
199  // current network-enforced consensus rules should be used. In
200  // a future soft-fork scenario that would mean checking which
201  // rules would be enforced for the next block and setting the
202  // appropriate flags. At the present time no soft-forks are
203  // scheduled, so no flags are set.
204  flags = std::max(flags, 0);
205 
206  // CheckFinalTx() uses chainActive.Height()+1 to evaluate
207  // nLockTime because when IsFinalTx() is called within
208  // CBlock::AcceptBlock(), the height of the block *being*
209  // evaluated is what is used. Thus if we want to know if a
210  // transaction can be part of the *next* block, we need to call
211  // IsFinalTx() with one more than chainActive.Height().
212  const int nBlockHeight = chainActive.Height() + 1;
213 
214  // BIP113 will require that time-locked transactions have nLockTime set to
215  // less than the median time of the previous block they're contained in.
216  // When the next block is created its previous block will be the current
217  // chain tip, so we use that to calculate the median time passed to
218  // IsFinalTx() if LOCKTIME_MEDIAN_TIME_PAST is set.
219  const int64_t nBlockTime = (flags & LOCKTIME_MEDIAN_TIME_PAST)
220  ? chainActive.Tip()->GetMedianTimePast()
221  : GetAdjustedTime();
222 
223  return IsFinalTx(tx, nBlockHeight, nBlockTime);
224 }

244 bool CheckSequenceLocks(const CTransaction &tx, int flags, LockPoints* lp, bool useExistingLockPoints)
245 {
246  AssertLockHeld(cs_main);
247  AssertLockHeld(mempool.cs);
248 
249  CBlockIndex* tip = chainActive.Tip();
250  CBlockIndex index;
251  index.pprev = tip;
252  // CheckSequenceLocks() uses chainActive.Height()+1 to evaluate
253  // height based locks because when SequenceLocks() is called within
254  // ConnectBlock(), the height of the block *being*
255  // evaluated is what is used.
256  // Thus if we want to know if a transaction can be part of the
257  // *next* block, we need to use one more than chainActive.Height()
258  index.nHeight = tip->nHeight + 1;
259 
260  std::pair<int, int64_t> lockPair;
261  if (useExistingLockPoints) {
262  assert(lp);
263  lockPair.first = lp->height;
264  lockPair.second = lp->time;
265  }
266  else {
267  // pcoinsTip contains the UTXO set for chainActive.Tip()
268  CCoinsViewMemPool viewMemPool(pcoinsTip, mempool);
269  std::vector<int> prevheights;
270  prevheights.resize(tx.vin.size());
271  for (size_t txinIndex = 0; txinIndex < tx.vin.size(); txinIndex++) {
272  const CTxIn& txin = tx.vin[txinIndex];
273  Coin coin;
274  if (!viewMemPool.GetCoin(txin.prevout, coin)) {
275  return error("%s: Missing input", __func__);
276  }
277  if (coin.nHeight == MEMPOOL_HEIGHT) {
278  // Assume all mempool transaction confirm in the next block
279  prevheights[txinIndex] = tip->nHeight + 1;
280  } else {
281  prevheights[txinIndex] = coin.nHeight;
282  }
283  }
284  lockPair = CalculateSequenceLocks(tx, flags, &prevheights, index);
285  if (lp) {
286  lp->height = lockPair.first;
287  lp->time = lockPair.second;
288  // Also store the hash of the block with the highest height of
289  // all the blocks which have sequence locked prevouts.
290  // This hash needs to still be on the chain
291  // for these LockPoint calculations to be valid
292  // Note: It is impossible to correctly calculate a maxInputBlock
293  // if any of the sequence locked inputs depend on unconfirmed txs,
294  // except in the special case where the relative lock time/height
295  // is 0, which is equivalent to no sequence lock. Since we assume
296  // input height of tip+1 for mempool txs and test the resulting
297  // lockPair from CalculateSequenceLocks against tip+1. We know
298  // EvaluateSequenceLocks will fail if there was a non-zero sequence
299  // lock on a mempool input, so we can use the return value of
300  // CheckSequenceLocks to indicate the LockPoints validity
301  int maxInputHeight = 0;
302  BOOST_FOREACH(int height, prevheights) {
303  // Can ignore mempool inputs since we'll fail if they had non-zero locks
304  if (height != tip->nHeight+1) {
305  maxInputHeight = std::max(maxInputHeight, height);
306  }
307  }
308  lp->maxInputBlock = tip->GetAncestor(maxInputHeight);
309  }
310  }
311  return EvaluateSequenceLocks(index, lockPair);
312 }

347 /* Make mempool consistent after a reorg, by re-adding or recursively erasing
348  * disconnected block transactions from the mempool, and also removing any
349  * other transactions from the mempool that are no longer valid given the new
350  * tip/height.
351  *
352  * Note: we assume that disconnectpool only contains transactions that are NOT
353  * confirmed in the current chain nor already in the mempool (otherwise,
354  * in-mempool descendants of such transactions would be removed).
355  *
356  * Passing fAddToMempool=false will skip trying to add the transactions back,
357  * and instead just erase from the mempool as needed.
358  */
359 
360 void UpdateMempoolForReorg(DisconnectedBlockTransactions &disconnectpool, bool fAddToMempool)
361 {
362  AssertLockHeld(cs_main);
363  std::vector<uint256> vHashUpdate;
364  // disconnectpool's insertion_order index sorts the entries from
365  // oldest to newest, but the oldest entry will be the last tx from the
366  // latest mined block that was disconnected.
367  // Iterate disconnectpool in reverse, so that we add transactions
368  // back to the mempool starting with the earliest transaction that had
369  // been previously seen in a block.
370  auto it = disconnectpool.queuedTx.get<insertion_order>().rbegin();
371  while (it != disconnectpool.queuedTx.get<insertion_order>().rend()) {
372  // ignore validation errors in resurrected transactions
373  CValidationState stateDummy;
374  if (!fAddToMempool || (*it)->IsCoinBase() || !AcceptToMemoryPool(mempool, stateDummy, *it, false, NULL, NULL, true)) {
375  // If the transaction doesn't make it in to the mempool, remove any
376  // transactions that depend on it (which would now be orphans).
377  mempool.removeRecursive(**it, MemPoolRemovalReason::REORG);
378  } else if (mempool.exists((*it)->GetHash())) {
379  vHashUpdate.push_back((*it)->GetHash());
380  }
381  ++it;
382  }
383  disconnectpool.queuedTx.clear();
384  // AcceptToMemoryPool/addUnchecked all assume that new mempool entries have
385  // no in-mempool children, which is generally not true when adding
386  // previously-confirmed transactions back to the mempool.
387  // UpdateTransactionsFromBlock finds descendants of any transactions in
388  // the disconnectpool that were added back and cleans up the mempool state.
389  mempool.UpdateTransactionsFromBlock(vHashUpdate);
390 
391  // We also need to remove any now-immature transactions
392  mempool.removeForReorg(pcoinsTip, chainActive.Tip()->nHeight + 1, STANDARD_LOCKTIME_VERIFY_FLAGS);
393  // Re-limit mempool size, in case we added any transactions
394  LimitMempoolSize(mempool, GetArg("-maxmempool", DEFAULT_MAX_MEMPOOL_SIZE) * 1000000, GetArg("-mempoolexpiry", DEFAULT_MEMPOOL_EXPIRY) * 60 * 60);
395 }

397 static bool AcceptToMemoryPoolWorker(const CChainParams& chainparams, CTxMemPool& pool, CValidationState& state, const CTransactionRef& ptx, bool fLimitFree,
398  bool* pfMissingInputs, int64_t nAcceptTime, std::list<CTransactionRef>* plTxnReplaced,
399  bool fOverrideMempoolLimit, const CAmount& nAbsurdFee, std::vector<COutPoint>& coins_to_uncache)
400 {
401  const CTransaction& tx = *ptx;
402  const uint256 hash = tx.GetHash();
403  AssertLockHeld(cs_main);
404  if (pfMissingInputs)
405  *pfMissingInputs = false;
406 
407  if (!CheckTransaction(tx, state))
408  return false; // state filled in by CheckTransaction
409 
410  // Coinbase is only valid in a block, not as a loose transaction
411  if (tx.IsCoinBase())
412  return state.DoS(100, false, REJECT_INVALID, "coinbase");
413 
414  // Reject transactions with witness before segregated witness activates (override with -prematurewitness)
415  bool witnessEnabled = IsWitnessEnabled(chainActive.Tip(), chainparams.GetConsensus());
416  if (!GetBoolArg("-prematurewitness",false) && tx.HasWitness() && !witnessEnabled) {
417  return state.DoS(0, false, REJECT_NONSTANDARD, "no-witness-yet", true);
418  }
419 
420  // Rather not work on nonstandard transactions (unless -testnet/-regtest)
421  std::string reason;
422  if (fRequireStandard && !IsStandardTx(tx, reason, witnessEnabled))
423  return state.DoS(0, false, REJECT_NONSTANDARD, reason);
424 
425  // Only accept nLockTime-using transactions that can be mined in the next
426  // block; we don't want our mempool filled up with transactions that can't
427  // be mined yet.
428  if (!CheckFinalTx(tx, STANDARD_LOCKTIME_VERIFY_FLAGS))
429  return state.DoS(0, false, REJECT_NONSTANDARD, "non-final");
430 
431  // is it already in the memory pool?
432  if (pool.exists(hash))
433  return state.Invalid(false, REJECT_ALREADY_KNOWN, "txn-already-in-mempool");
434 
435  // Check for conflicts with in-memory transactions
436  std::set<uint256> setConflicts;
437  {
438  LOCK(pool.cs); // protect pool.mapNextTx
439  BOOST_FOREACH(const CTxIn &txin, tx.vin)
440  {
441  auto itConflicting = pool.mapNextTx.find(txin.prevout);
442  if (itConflicting != pool.mapNextTx.end())
443  {
444  const CTransaction *ptxConflicting = itConflicting->second;
445  if (!setConflicts.count(ptxConflicting->GetHash()))
446  {
447  // Allow opt-out of transaction replacement by setting
448  // nSequence >= maxint-1 on all inputs.
449  //
450  // maxint-1 is picked to still allow use of nLockTime by
451  // non-replaceable transactions. All inputs rather than just one
452  // is for the sake of multi-party protocols, where we don't
453  // want a single party to be able to disable replacement.
454  //
455  // The opt-out ignores descendants as anyone relying on
456  // first-seen mempool behavior should be checking all
457  // unconfirmed ancestors anyway; doing otherwise is hopelessly
458  // insecure.
459  bool fReplacementOptOut = true;
460  if (fEnableReplacement)
461  {
462  BOOST_FOREACH(const CTxIn &_txin, ptxConflicting->vin)
463  {
464  if (_txin.nSequence < std::numeric_limits<unsigned int>::max()-1)
465  {
466  fReplacementOptOut = false;
467  break;
468  }
469  }
470  }
471  if (fReplacementOptOut)
472  return state.Invalid(false, REJECT_CONFLICT, "txn-mempool-conflict");
473 
474  setConflicts.insert(ptxConflicting->GetHash());
475  }
476  }
477  }
478  }
479 
480  {
481  CCoinsView dummy;
482  CCoinsViewCache view(&dummy);
483 
484  CAmount nValueIn = 0;
485  LockPoints lp;
486  {
487  LOCK(pool.cs);
488  CCoinsViewMemPool viewMemPool(pcoinsTip, pool);
489  view.SetBackend(viewMemPool);
490 
491  // do we already have it?
492  for (size_t out = 0; out < tx.vout.size(); out++) {
493  COutPoint outpoint(hash, out);
494  bool had_coin_in_cache = pcoinsTip->HaveCoinInCache(outpoint);
495  if (view.HaveCoin(outpoint)) {
496  if (!had_coin_in_cache) {
497  coins_to_uncache.push_back(outpoint);
498  }
499  return state.Invalid(false, REJECT_ALREADY_KNOWN, "txn-already-known");
500  }
501  }
502 
503  // do all inputs exist?
504  BOOST_FOREACH(const CTxIn txin, tx.vin) {
505  if (!pcoinsTip->HaveCoinInCache(txin.prevout)) {
506  coins_to_uncache.push_back(txin.prevout);
507  }
508  if (!view.HaveCoin(txin.prevout)) {
509  if (pfMissingInputs) {
510  *pfMissingInputs = true;
511  }
512  return false; // fMissingInputs and !state.IsInvalid() is used to detect this condition, don't set state.Invalid()
513  }
514  }
515 
516  // Bring the best block into scope
517  view.GetBestBlock();
518 
519  nValueIn = view.GetValueIn(tx);
520 
521  // we have all inputs cached now, so switch back to dummy, so we don't need to keep lock on mempool
522  view.SetBackend(dummy);
523 
524  // Only accept BIP68 sequence locked transactions that can be mined in the next
525  // block; we don't want our mempool filled up with transactions that can't
526  // be mined yet.
527  // Must keep pool.cs for this unless we change CheckSequenceLocks to take a
528  // CoinsViewCache instead of create its own
529  if (!CheckSequenceLocks(tx, STANDARD_LOCKTIME_VERIFY_FLAGS, &lp))
530  return state.DoS(0, false, REJECT_NONSTANDARD, "non-BIP68-final");
531  }
532 
533  // Check for non-standard pay-to-script-hash in inputs
534  if (fRequireStandard && !AreInputsStandard(tx, view))
535  return state.Invalid(false, REJECT_NONSTANDARD, "bad-txns-nonstandard-inputs");
536 
537  // Check for non-standard witness in P2WSH
538  if (tx.HasWitness() && fRequireStandard && !IsWitnessStandard(tx, view))
539  return state.DoS(0, false, REJECT_NONSTANDARD, "bad-witness-nonstandard", true);
540 
541  int64_t nSigOpsCost = GetTransactionSigOpCost(tx, view, STANDARD_SCRIPT_VERIFY_FLAGS);
542 
543  CAmount nValueOut = tx.GetValueOut();
544  CAmount nFees = nValueIn-nValueOut;
545  // nModifiedFees includes any fee deltas from PrioritiseTransaction
546  CAmount nModifiedFees = nFees;
547  pool.ApplyDelta(hash, nModifiedFees);
548 
549  // Keep track of transactions that spend a coinbase, which we re-scan
550  // during reorgs to ensure COINBASE_MATURITY is still met.
551  bool fSpendsCoinbase = false;
552  BOOST_FOREACH(const CTxIn &txin, tx.vin) {
553  const Coin &coin = view.AccessCoin(txin.prevout);
554  if (coin.IsCoinBase()) {
555  fSpendsCoinbase = true;
556  break;
557  }
558  }
559 
560  CTxMemPoolEntry entry(ptx, nFees, nAcceptTime, chainActive.Height(),
561  fSpendsCoinbase, nSigOpsCost, lp);
562  unsigned int nSize = entry.GetTxSize();
563 
564  // Check that the transaction doesn't have an excessive number of
565  // sigops, making it impossible to mine. Since the coinbase transaction
566  // itself can contain sigops MAX_STANDARD_TX_SIGOPS is less than
567  // MAX_BLOCK_SIGOPS; we still consider this an invalid rather than
568  // merely non-standard transaction.
569  if (nSigOpsCost > MAX_STANDARD_TX_SIGOPS_COST)
570  return state.DoS(0, false, REJECT_NONSTANDARD, "bad-txns-too-many-sigops", false,
571  strprintf("%d", nSigOpsCost));
572 
573  CAmount mempoolRejectFee = pool.GetMinFee(GetArg("-maxmempool", DEFAULT_MAX_MEMPOOL_SIZE) * 1000000).GetFee(nSize);
574  if (mempoolRejectFee > 0 && nModifiedFees < mempoolRejectFee) {
575  return state.DoS(0, false, REJECT_INSUFFICIENTFEE, "mempool min fee not met", false, strprintf("%d < %d", nFees, mempoolRejectFee));
576  }
577 
578  // No transactions are allowed below minRelayTxFee except from disconnected blocks
579  if (fLimitFree && nModifiedFees < ::minRelayTxFee.GetFee(nSize)) {
580  return state.DoS(0, false, REJECT_INSUFFICIENTFEE, "min relay fee not met");
581  }
582 
583  if (nAbsurdFee && nFees > nAbsurdFee)
584  return state.Invalid(false,
585  REJECT_HIGHFEE, "absurdly-high-fee",
586  strprintf("%d > %d", nFees, nAbsurdFee));
587 
588  // Calculate in-mempool ancestors, up to a limit.
589  CTxMemPool::setEntries setAncestors;
590  size_t nLimitAncestors = GetArg("-limitancestorcount", DEFAULT_ANCESTOR_LIMIT);
591  size_t nLimitAncestorSize = GetArg("-limitancestorsize", DEFAULT_ANCESTOR_SIZE_LIMIT)*1000;
592  size_t nLimitDescendants = GetArg("-limitdescendantcount", DEFAULT_DESCENDANT_LIMIT);
593  size_t nLimitDescendantSize = GetArg("-limitdescendantsize", DEFAULT_DESCENDANT_SIZE_LIMIT)*1000;
594  std::string errString;
595  if (!pool.CalculateMemPoolAncestors(entry, setAncestors, nLimitAncestors, nLimitAncestorSize, nLimitDescendants, nLimitDescendantSize, errString)) {
596  return state.DoS(0, false, REJECT_NONSTANDARD, "too-long-mempool-chain", false, errString);
597  }
598 
599  // A transaction that spends outputs that would be replaced by it is invalid. Now
600  // that we have the set of all ancestors we can detect this
601  // pathological case by making sure setConflicts and setAncestors don't
602  // intersect.
603  BOOST_FOREACH(CTxMemPool::txiter ancestorIt, setAncestors)
604  {
605  const uint256 &hashAncestor = ancestorIt->GetTx().GetHash();
606  if (setConflicts.count(hashAncestor))
607  {
608  return state.DoS(10, false,
609  REJECT_INVALID, "bad-txns-spends-conflicting-tx", false,
610  strprintf("%s spends conflicting transaction %s",
611  hash.ToString(),
612  hashAncestor.ToString()));
613  }
614  }
615 
616  // Check if it's economically rational to mine this transaction rather
617  // than the ones it replaces.
618  CAmount nConflictingFees = 0;
619  size_t nConflictingSize = 0;
620  uint64_t nConflictingCount = 0;
621  CTxMemPool::setEntries allConflicting;
622 
623  // If we don't hold the lock allConflicting might be incomplete; the
624  // subsequent RemoveStaged() and addUnchecked() calls don't guarantee
625  // mempool consistency for us.
626  LOCK(pool.cs);
627  const bool fReplacementTransaction = setConflicts.size();
628  if (fReplacementTransaction)
629  {
630  CFeeRate newFeeRate(nModifiedFees, nSize);
631  std::set<uint256> setConflictsParents;
632  const int maxDescendantsToVisit = 100;
633  CTxMemPool::setEntries setIterConflicting;
634  BOOST_FOREACH(const uint256 &hashConflicting, setConflicts)
635  {
636  CTxMemPool::txiter mi = pool.mapTx.find(hashConflicting);
637  if (mi == pool.mapTx.end())
638  continue;
639 
640  // Save these to avoid repeated lookups
641  setIterConflicting.insert(mi);
642 
643  // Don't allow the replacement to reduce the feerate of the
644  // mempool.
645  //
646  // We usually don't want to accept replacements with lower
647  // feerates than what they replaced as that would lower the
648  // feerate of the next block. Requiring that the feerate always
649  // be increased is also an easy-to-reason about way to prevent
650  // DoS attacks via replacements.
651  //
652  // The mining code doesn't (currently) take children into
653  // account (CPFP) so we only consider the feerates of
654  // transactions being directly replaced, not their indirect
655  // descendants. While that does mean high feerate children are
656  // ignored when deciding whether or not to replace, we do
657  // require the replacement to pay more overall fees too,
658  // mitigating most cases.
659  CFeeRate oldFeeRate(mi->GetModifiedFee(), mi->GetTxSize());
660  if (newFeeRate <= oldFeeRate)
661  {
662  return state.DoS(0, false,
663  REJECT_INSUFFICIENTFEE, "insufficient fee", false,
664  strprintf("rejecting replacement %s; new feerate %s <= old feerate %s",
665  hash.ToString(),
666  newFeeRate.ToString(),
667  oldFeeRate.ToString()));
668  }
669 
670  BOOST_FOREACH(const CTxIn &txin, mi->GetTx().vin)
671  {
672  setConflictsParents.insert(txin.prevout.hash);
673  }
674 
675  nConflictingCount += mi->GetCountWithDescendants();
676  }
677  // This potentially overestimates the number of actual descendants
678  // but we just want to be conservative to avoid doing too much
679  // work.
680  if (nConflictingCount <= maxDescendantsToVisit) {
681  // If not too many to replace, then calculate the set of
682  // transactions that would have to be evicted
683  BOOST_FOREACH(CTxMemPool::txiter it, setIterConflicting) {
684  pool.CalculateDescendants(it, allConflicting);
685  }
686  BOOST_FOREACH(CTxMemPool::txiter it, allConflicting) {
687  nConflictingFees += it->GetModifiedFee();
688  nConflictingSize += it->GetTxSize();
689  }
690  } else {
691  return state.DoS(0, false,
692  REJECT_NONSTANDARD, "too many potential replacements", false,
693  strprintf("rejecting replacement %s; too many potential replacements (%d > %d)\n",
694  hash.ToString(),
695  nConflictingCount,
696  maxDescendantsToVisit));
697  }
698 
699  for (unsigned int j = 0; j < tx.vin.size(); j++)
700  {
701  // We don't want to accept replacements that require low
702  // feerate junk to be mined first. Ideally we'd keep track of
703  // the ancestor feerates and make the decision based on that,
704  // but for now requiring all new inputs to be confirmed works.
705  if (!setConflictsParents.count(tx.vin[j].prevout.hash))
706  {
707  // Rather than check the UTXO set - potentially expensive -
708  // it's cheaper to just check if the new input refers to a
709  // tx that's in the mempool.
710  if (pool.mapTx.find(tx.vin[j].prevout.hash) != pool.mapTx.end())
711  return state.DoS(0, false,
712  REJECT_NONSTANDARD, "replacement-adds-unconfirmed", false,
713  strprintf("replacement %s adds unconfirmed input, idx %d",
714  hash.ToString(), j));
715  }
716  }
717 
718  // The replacement must pay greater fees than the transactions it
719  // replaces - if we did the bandwidth used by those conflicting
720  // transactions would not be paid for.
721  if (nModifiedFees < nConflictingFees)
722  {
723  return state.DoS(0, false,
724  REJECT_INSUFFICIENTFEE, "insufficient fee", false,
725  strprintf("rejecting replacement %s, less fees than conflicting txs; %s < %s",
726  hash.ToString(), FormatMoney(nModifiedFees), FormatMoney(nConflictingFees)));
727  }
728 
729  // Finally in addition to paying more fees than the conflicts the
730  // new transaction must pay for its own bandwidth.
731  CAmount nDeltaFees = nModifiedFees - nConflictingFees;
732  if (nDeltaFees < ::incrementalRelayFee.GetFee(nSize))
733  {
734  return state.DoS(0, false,
735  REJECT_INSUFFICIENTFEE, "insufficient fee", false,
736  strprintf("rejecting replacement %s, not enough additional fees to relay; %s < %s",
737  hash.ToString(),
738  FormatMoney(nDeltaFees),
739  FormatMoney(::incrementalRelayFee.GetFee(nSize))));
740  }
741  }
742 
743  unsigned int scriptVerifyFlags = STANDARD_SCRIPT_VERIFY_FLAGS;
744  if (!chainparams.RequireStandard()) {
745  scriptVerifyFlags = GetArg("-promiscuousmempoolflags", scriptVerifyFlags);
746  }
747 
748  // Check against previous transactions
749  // This is done last to help prevent CPU exhaustion denial-of-service attacks.
750  PrecomputedTransactionData txdata(tx);
751  if (!CheckInputs(tx, state, view, true, scriptVerifyFlags, true, txdata)) {
752  // SCRIPT_VERIFY_CLEANSTACK requires SCRIPT_VERIFY_WITNESS, so we
753  // need to turn both off, and compare against just turning off CLEANSTACK
754  // to see if the failure is specifically due to witness validation.
755  CValidationState stateDummy; // Want reported failures to be from first CheckInputs
756  if (!tx.HasWitness() && CheckInputs(tx, stateDummy, view, true, scriptVerifyFlags & ~(SCRIPT_VERIFY_WITNESS | SCRIPT_VERIFY_CLEANSTACK), true, txdata) &&
757  !CheckInputs(tx, stateDummy, view, true, scriptVerifyFlags & ~SCRIPT_VERIFY_CLEANSTACK, true, txdata)) {
758  // Only the witness is missing, so the transaction itself may be fine.
759  state.SetCorruptionPossible();
760  }
761  return false; // state filled in by CheckInputs
762  }
763 
764  // Check again against just the consensus-critical mandatory script
765  // verification flags, in case of bugs in the standard flags that cause
766  // transactions to pass as valid when they're actually invalid. For
767  // instance the STRICTENC flag was incorrectly allowing certain
768  // CHECKSIG NOT scripts to pass, even though they were invalid.
769  //
770  // There is a similar check in CreateNewBlock() to prevent creating
771  // invalid blocks, however allowing such transactions into the mempool
772  // can be exploited as a DoS attack.
773  if (!CheckInputs(tx, state, view, true, MANDATORY_SCRIPT_VERIFY_FLAGS, true, txdata))
774  {
775  return error("%s: BUG! PLEASE REPORT THIS! ConnectInputs failed against MANDATORY but not STANDARD flags %s, %s",
776  __func__, hash.ToString(), FormatStateMessage(state));
777  }
778 
779  // Remove conflicting transactions from the mempool
780  BOOST_FOREACH(const CTxMemPool::txiter it, allConflicting)
781  {
782  LogPrint(BCLog::MEMPOOL, "replacing tx %s with %s for %s ION additional fees, %d delta bytes\n",
783  it->GetTx().GetHash().ToString(),
784  hash.ToString(),
785  FormatMoney(nModifiedFees - nConflictingFees),
786  (int)nSize - (int)nConflictingSize);
787  if (plTxnReplaced)
788  plTxnReplaced->push_back(it->GetSharedTx());
789  }
790  pool.RemoveStaged(allConflicting, false, MemPoolRemovalReason::REPLACED);
791 
792  // This transaction should only count for fee estimation if it isn't a
793  // BIP 125 replacement transaction (may not be widely supported), the
794  // node is not behind, and the transaction is not dependent on any other
795  // transactions in the mempool.
796  bool validForFeeEstimation = !fReplacementTransaction && IsCurrentForFeeEstimation() && pool.HasNoInputsOf(tx);
797 
798  // Store transaction in memory
799  pool.addUnchecked(hash, entry, setAncestors, validForFeeEstimation);
800 
801  // trim mempool and check if tx was trimmed
802  if (!fOverrideMempoolLimit) {
803  LimitMempoolSize(pool, GetArg("-maxmempool", DEFAULT_MAX_MEMPOOL_SIZE) * 1000000, GetArg("-mempoolexpiry", DEFAULT_MEMPOOL_EXPIRY) * 60 * 60);
804  if (!pool.exists(hash))
805  return state.DoS(0, false, REJECT_INSUFFICIENTFEE, "mempool full");
806  }
807  }
808 
809  GetMainSignals().TransactionAddedToMempool(ptx);
810 
811  return true;
812 }

1169 
1170  // Skip script verification when connecting blocks under the
1171  // assumevalid block. Assuming the assumevalid block is valid this
1172  // is safe because block merkle hashes are still computed and checked,
1173  // Of course, if an assumed valid block is invalid due to false scriptSigs
1174  // this optimization would allow an invalid chain to be accepted.
1175  if (fScriptChecks) {
1176  for (unsigned int i = 0; i < tx.vin.size(); i++) {
1177  const COutPoint &prevout = tx.vin[i].prevout;
1178  const Coin& coin = inputs.AccessCoin(prevout);
1179  assert(!coin.IsSpent());
1180 
1181  // We very carefully only pass in things to CScriptCheck which
1182  // are clearly committed to by tx' witness hash. This provides
1183  // a sanity check that our caching is not introducing consensus
1184  // failures through additional data in, eg, the coins being
1185  // spent being checked as a part of CScriptCheck.
1186  const CScript& scriptPubKey = coin.out.scriptPubKey;
1187  const CAmount amount = coin.out.nValue;
1188 
1189  // Verify signature
1190  CScriptCheck check(scriptPubKey, amount, tx, i, flags, cacheStore, &txdata);
1191  if (pvChecks) {
1192  pvChecks->push_back(CScriptCheck());
1193  check.swap(pvChecks->back());
1194  } else if (!check()) {
1195  if (flags & STANDARD_NOT_MANDATORY_VERIFY_FLAGS) {
1196  // Check whether the failure was caused by a
1197  // non-mandatory script verification check, such as
1198  // non-standard DER encodings or non-null dummy
1199  // arguments; if so, don't trigger DoS protection to
1200  // avoid splitting the network between upgraded and
1201  // non-upgraded nodes.
1202  CScriptCheck check2(scriptPubKey, amount, tx, i,
1203  flags & ~STANDARD_NOT_MANDATORY_VERIFY_FLAGS, cacheStore, &txdata);
1204  if (check2())
1205  return state.Invalid(false, REJECT_NONSTANDARD, strprintf("non-mandatory-script-verify-flag (%s)", ScriptErrorString(check.GetScriptError())));
1206  }
1207  // Failures of other flags indicate a transaction that is
1208  // invalid in new blocks, e.g. a invalid P2SH. We DoS ban
1209  // such nodes as they are not following the protocol. That
1210  // said during an upgrade careful thought should be taken
1211  // as to the correct behavior - we may want to continue
1212  // peering with non-upgraded nodes even after soft-fork
1213  // super-majority signaling has occurred.
1214  return state.DoS(100,false, REJECT_INVALID, strprintf("mandatory-script-verify-flag-failed (%s)", ScriptErrorString(check.GetScriptError())));
1215  }
1216  }
1217  }
1218  }
1219 
1220  return true;

1337 static DisconnectResult DisconnectBlock(const CBlock& block, const CBlockIndex* pindex, CCoinsViewCache& view)
1338 {
1339  assert(pindex->GetBlockHash() == view.GetBestBlock());
1340 
1341  bool fClean = true;
1342 
1343  CBlockUndo blockUndo;
1344  CDiskBlockPos pos = pindex->GetUndoPos();
1345  if (pos.IsNull()) {
1346  error("DisconnectBlock(): no undo data available");
1347  return DISCONNECT_FAILED;
1348  }
1349  if (!UndoReadFromDisk(blockUndo, pos, pindex->pprev->GetBlockHash())) {
1350  error("DisconnectBlock(): failure reading undo data");
1351  return DISCONNECT_FAILED;
1352  }
1353 
1354  if (blockUndo.vtxundo.size() + 1 != block.vtx.size()) {
1355  error("DisconnectBlock(): block and undo data inconsistent");
1356  return DISCONNECT_FAILED;
1357  }
1358 
1359  // undo transactions in reverse order
1360  for (int i = block.vtx.size() - 1; i >= 0; i--) {
1361  const CTransaction &tx = *(block.vtx[i]);
1362  uint256 hash = tx.GetHash();
1363 
1364  // Check that all outputs are available and match the outputs in the block itself
1365  // exactly.
1366  for (size_t o = 0; o < tx.vout.size(); o++) {
1367  if (!tx.vout[o].scriptPubKey.IsUnspendable()) {
1368  COutPoint out(hash, o);
1369  Coin coin;
1370  view.SpendCoin(out, &coin);
1371  if (tx.vout[o] != coin.out) {
1372  fClean = false; // transaction output mismatch
1373  }
1374  }
1375  }
1376 
1377  // restore inputs
1378  if (i > 0) { // not coinbases
1379  CTxUndo &txundo = blockUndo.vtxundo[i-1];
1380  if (txundo.vprevout.size() != tx.vin.size()) {
1381  error("DisconnectBlock(): transaction and undo data inconsistent");
1382  return DISCONNECT_FAILED;
1383  }
1384  for (unsigned int j = tx.vin.size(); j-- > 0;) {
1385  const COutPoint &out = tx.vin[j].prevout;
1386  int res = ApplyTxInUndo(std::move(txundo.vprevout[j]), view, out);
1387  if (res == DISCONNECT_FAILED) return DISCONNECT_FAILED;
1388  fClean = fClean && res != DISCONNECT_UNCLEAN;
1389  }
1390  // At this point, all of txundo.vprevout should have been moved out.
1391  }
1392  }
1393 
1394  // move best block pointer to prevout block
1395  view.SetBestBlock(pindex->pprev->GetBlockHash());
1396 
1397  return fClean ? DISCONNECT_OK : DISCONNECT_UNCLEAN;
1398 }

1496  int64_t nTimeStart = GetTimeMicros();
1497

1499  if (!CheckBlock(block, state, chainparams.GetConsensus(), !fJustCheck, !fJustCheck))
1500  return error("%s: Consensus::CheckBlock: %s", __func__, FormatStateMessage(state));
1501 
1502  // verify that the view's current state corresponds to the previous block
1503  uint256 hashPrevBlock = pindex->pprev == NULL ? uint256() : pindex->pprev->GetBlockHash();
1504  assert(hashPrevBlock == view.GetBestBlock());

1506  // Special case for the genesis block, skipping connection of its transactions
1507  // (its coinbase is unspendable)
1508  if (block.GetHash() == chainparams.GetConsensus().hashGenesisBlock) {
1509  if (!fJustCheck)
1510  view.SetBestBlock(pindex->GetBlockHash());

1512  }
1513 
1514  bool fScriptChecks = true;
1515  if (!hashAssumeValid.IsNull()) {
1516  // We've been configured with the hash of a block which has been externally verified to have a valid history.
1517  // A suitable default value is included with the software and updated from time to time. Because validity
1518  // relative to a piece of software is an objective fact these defaults can be easily reviewed.
1519  // This setting doesn't force the selection of any particular chain but makes validating some faster by
1520  // effectively caching the result of part of the verification.
1521  BlockMap::const_iterator it = mapBlockIndex.find(hashAssumeValid);
1522  if (it != mapBlockIndex.end()) {
1523  if (it->second->GetAncestor(pindex->nHeight) == pindex &&
1524  pindexBestHeader->GetAncestor(pindex->nHeight) == pindex &&
1525  pindexBestHeader->nChainWork >= UintToArith256(chainparams.GetConsensus().nMinimumChainWork)) {
1526  // This block is a member of the assumed verified chain and an ancestor of the best header.
1527  // The equivalent time check discourages hash power from extorting the network via DOS attack
1528  // into accepting an invalid block through telling users they must manually set assumevalid.
1529  // Requiring a software change or burying the invalid block, regardless of the setting, makes
1530  // it hard to hide the implication of the demand. This also avoids having release candidates
1531  // that are hardly doing any signature verification at all in testing without having to
1532  // artificially set the default assumed verified block further back.
1533  // The test against nMinimumChainWork prevents the skipping when denied access to any chain at
1534  // least as good as the expected chain.

1543  // Do not allow blocks that contain transactions which 'overwrite' older transactions,
1544  // unless those are already completely spent.
1545  // If such overwrites are allowed, coinbases and transactions depending upon those
1546  // can be duplicated to remove the ability to spend the first instance -- even after
1547  // being sent to another address.
1548  // See BIP30 and http://r6.ca/blog/20120206T005236Z.html for more information.
1549  // This logic is not necessary for memory pool transactions, as AcceptToMemoryPool
1550  // already refuses previously-known transaction ids entirely.
1551  // This rule was originally applied to all blocks with a timestamp after March 15, 2012, 0:00 UTC.
1552  // Now that the whole chain is irreversibly beyond that time it is applied to all blocks except the

1562  // before the first had been spent. Since those coinbases are sufficiently buried its no longer possible to create further
1563  // duplicate transactions descending from the known pairs either.

1787  bool fPeriodicFlush = mode == FLUSH_STATE_PERIODIC && nNow > nLastFlush + (int64_t)DATABASE_FLUSH_INTERVAL * 1000000;

1836  GetMainSignals().SetBestChain(chainActive.GetLocator());

1838  }
1839  } catch (const std::runtime_error& e) {

1841  }

2853  // * We build a merkle tree with all those witness hashes as leaves (similar to the hashMerkleRoot in the block header).
2854  // * There must be at least one output whose scriptPubKey is a single 36-byte push, the first 4 bytes of which are

2856  // multiple, the last one is used.

2868  }

2871  return state.DoS(100, false, REJECT_INVALID, "bad-witness-merkle-match", true, strprintf("%s : witness merkle commitment mismatch", __func__));

2874  }
2875  }

2882  }
2883  }
2884  }

2887  // we can check if the block weight passes (before we've checked the

2890  // the block hash, so we couldn't mark the block as permanently
