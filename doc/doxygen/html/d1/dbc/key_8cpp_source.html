32 static int ec_privkey_import_der(const secp256k1_context* ctx, unsigned char *out32, const unsigned char *privkey, size_t privkeylen) {
33  const unsigned char *end = privkey + privkeylen;
34  int lenb = 0;
35  int len = 0;
36  memset(out32, 0, 32);
37  /* sequence header */
38  if (end < privkey+1 || *privkey != 0x30) {
39  return 0;
40  }
41  privkey++;
42  /* sequence length constructor */
43  if (end < privkey+1 || !(*privkey & 0x80)) {
44  return 0;
45  }
46  lenb = *privkey & ~0x80; privkey++;
47  if (lenb < 1 || lenb > 2) {
48  return 0;
49  }
50  if (end < privkey+lenb) {
51  return 0;
52  }
53  /* sequence length */
54  len = privkey[lenb-1] | (lenb > 1 ? privkey[lenb-2] << 8 : 0);
55  privkey += lenb;
56  if (end < privkey+len) {
57  return 0;
58  }
59  /* sequence element 0: version number (=1) */
60  if (end < privkey+3 || privkey[0] != 0x02 || privkey[1] != 0x01 || privkey[2] != 0x01) {
61  return 0;
62  }
63  privkey += 3;
64  /* sequence element 1: octet string, up to 32 bytes */
65  if (end < privkey+2 || privkey[0] != 0x04 || privkey[1] > 0x20 || end < privkey+2+privkey[1]) {
66  return 0;
67  }
68  memcpy(out32 + 32 - privkey[1], privkey + 2, privkey[1]);
69  if (!secp256k1_ec_seckey_verify(ctx, out32)) {
70  memset(out32, 0, 32);
71  return 0;
72  }
73  return 1;
74 }

238 bool CKey::Derive(CKey& keyChild, unsigned char ccChild[32], unsigned int nChild, const unsigned char cc[32]) const {
239  assert(IsValid());
240  assert(IsCompressed());
241  unsigned char out[64];
242  LockObject(out);
243  if ((nChild >> 31) == 0) {
244  CPubKey pubkey = GetPubKey();
245  assert(pubkey.begin() + 33 == pubkey.end());
246  BIP32Hash(cc, nChild, *pubkey.begin(), pubkey.begin()+1, out);
247  } else {
248  assert(begin() + 32 == end());
249  BIP32Hash(cc, nChild, 0, begin(), out);
250  }

252  memcpy((unsigned char*)keyChild.begin(), begin(), 32);

254  UnlockObject(out);
255  keyChild.fCompressed = true;
256  keyChild.fValid = ret;
257  return ret;
