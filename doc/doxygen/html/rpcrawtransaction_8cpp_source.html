51 void TxToJSON(const CTransaction& tx, const uint256 hashBlock, Object& entry)
52 {
53  entry.push_back(Pair("txid", tx.GetHash().GetHex()));
54  entry.push_back(Pair("version", tx.nVersion));
55  entry.push_back(Pair("time", (int64_t)tx.nTime));
56  entry.push_back(Pair("locktime", (int64_t)tx.nLockTime));
57  Array vin;
58  BOOST_FOREACH(const CTxIn& txin, tx.vin)
59  {
60  Object in;
61  if (tx.IsCoinBase())
62  in.push_back(Pair("coinbase", HexStr(txin.scriptSig.begin(), txin.scriptSig.end())));
63  else
64  {
65  in.push_back(Pair("txid", txin.prevout.hash.GetHex()));
66  in.push_back(Pair("vout", (int64_t)txin.prevout.n));
67  Object o;
68  o.push_back(Pair("asm", txin.scriptSig.ToString()));
69  o.push_back(Pair("hex", HexStr(txin.scriptSig.begin(), txin.scriptSig.end())));
70  in.push_back(Pair("scriptSig", o));
71  }
72  in.push_back(Pair("sequence", (int64_t)txin.nSequence));
73  vin.push_back(in);
74  }
75  entry.push_back(Pair("vin", vin));
76  Array vout;
77  for (unsigned int i = 0; i < tx.vout.size(); i++)
78  {
79  const CTxOut& txout = tx.vout[i];
80  Object out;
81  out.push_back(Pair("value", ValueFromAmount(txout.nValue)));
82  out.push_back(Pair("n", (int64_t)i));
83  Object o;
84  ScriptPubKeyToJSON(txout.scriptPubKey, o, true);
85  out.push_back(Pair("scriptPubKey", o));
86  vout.push_back(out);
87  }
88  entry.push_back(Pair("vout", vout));
89 
90  if (hashBlock != 0)
91  {
92  entry.push_back(Pair("blockhash", hashBlock.GetHex()));
93  map<uint256, CBlockIndex*>::iterator mi = mapBlockIndex.find(hashBlock);
94  if (mi != mapBlockIndex.end() && (*mi).second)
95  {
96  CBlockIndex* pindex = (*mi).second;
97  if (pindex->IsInMainChain())
98  {
99  entry.push_back(Pair("confirmations", 1 + nBestHeight - pindex->nHeight));
100  entry.push_back(Pair("time", (int64_t)pindex->nTime));
101  entry.push_back(Pair("blocktime", (int64_t)pindex->nTime));
102  }
103  else
104  entry.push_back(Pair("confirmations", 0));
105  }
106  }
107 }

175  if (setAddress.count(address))

195

221  entry.push_back(Pair("redeemScript", HexStr(redeemScript.begin(), redeemScript.end())));

225  entry.push_back(Pair("confirmations",out.nDepth));
226  entry.push_back(Pair("spendable", out.fSpendable));

228  }

347 Value signrawtransaction(const Array& params, bool fHelp)
348 {
349  if (fHelp || params.size() < 1 || params.size() > 4)
350  throw runtime_error(
351  "signrawtransaction <hex string> [{\"txid\":txid,\"vout\":n,\"scriptPubKey\":hex,\"redeemScript\":hex},...] [<privatekey1>,...] [sighashtype=\"ALL\"]\n"
352  "Sign inputs for raw transaction (serialized, hex-encoded).\n"
353  "Second optional argument (may be null) is an array of previous transaction outputs that\n"
354  "this transaction depends on but may not yet be in the blockchain.\n"
355  "Third optional argument (may be null) is an array of base58-encoded private\n"
356  "keys that, if given, will be the only keys used to sign the transaction.\n"
357  "Fourth optional argument is a string that is one of six values; ALL, NONE, SINGLE or\n"
358  "ALL|ANYONECANPAY, NONE|ANYONECANPAY, SINGLE|ANYONECANPAY.\n"
359  "Returns json object with keys:\n"
360  " hex : raw transaction with signature(s) (hex-encoded string)\n"
361  " complete : 1 if transaction has a complete set of signature (0 if not)"
362 #ifdef ENABLE_WALLET
363  + HelpRequiringPassphrase()
364 #endif
365  );
366 
367  RPCTypeCheck(params, list_of(str_type)(array_type)(array_type)(str_type), true);
368 
369  vector<unsigned char> txData(ParseHex(params[0].get_str()));
370  CDataStream ssData(txData, SER_NETWORK, PROTOCOL_VERSION);
371  vector<CTransaction> txVariants;
372  while (!ssData.empty())
373  {
374  try {
375  CTransaction tx;
376  ssData >> tx;
377  txVariants.push_back(tx);
378  }
379  catch (std::exception &e) {
380  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "TX decode failed");
381  }
382  }
383 
384  if (txVariants.empty())
385  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "Missing transaction");
386 
387  // mergedTx will end up with all the signatures; it
388  // starts as a clone of the rawtx:
389  CTransaction mergedTx(txVariants[0]);
390  bool fComplete = true;
391 
392  // Fetch previous transactions (inputs):
393  map<COutPoint, CScript> mapPrevOut;
394  for (unsigned int i = 0; i < mergedTx.vin.size(); i++)
395  {
396  CTransaction tempTx;
397  MapPrevTx mapPrevTx;
398  CTxDB txdb("r");
399  map<uint256, CTxIndex> unused;
400  bool fInvalid;
401 
402  // FetchInputs aborts on failure, so we go one at a time.
403  tempTx.vin.push_back(mergedTx.vin[i]);
404  tempTx.FetchInputs(txdb, unused, false, false, mapPrevTx, fInvalid);
405 
406  // Copy results into mapPrevOut:
407  BOOST_FOREACH(const CTxIn& txin, tempTx.vin)
408  {
409  const uint256& prevHash = txin.prevout.hash;
410  if (mapPrevTx.count(prevHash) && mapPrevTx[prevHash].second.vout.size()>txin.prevout.n)
411  mapPrevOut[txin.prevout] = mapPrevTx[prevHash].second.vout[txin.prevout.n].scriptPubKey;
412  }
413  }
414 
415  bool fGivenKeys = false;
416  CBasicKeyStore tempKeystore;
417  if (params.size() > 2 && params[2].type() != null_type)
418  {
419  fGivenKeys = true;
420  Array keys = params[2].get_array();
421  BOOST_FOREACH(Value k, keys)
422  {
423  CIonSecret vchSecret;
424  bool fGood = vchSecret.SetString(k.get_str());
425  if (!fGood)
426  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Invalid private key");
427  CKey key = vchSecret.GetKey();
428  tempKeystore.AddKey(key);
429  }
430  }
431 #ifdef ENABLE_WALLET
432  else
433  EnsureWalletIsUnlocked();
434 #endif
435 
436  // Add previous txouts given in the RPC call:
437  if (params.size() > 1 && params[1].type() != null_type)
438  {
439  Array prevTxs = params[1].get_array();
440  BOOST_FOREACH(Value& p, prevTxs)
441  {
442  if (p.type() != obj_type)
443  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "expected object with {\"txid'\",\"vout\",\"scriptPubKey\"}");
444 
445  Object prevOut = p.get_obj();
446 
447  RPCTypeCheck(prevOut, map_list_of("txid", str_type)("vout", int_type)("scriptPubKey", str_type));
448 
449  string txidHex = find_value(prevOut, "txid").get_str();
450  if (!IsHex(txidHex))
451  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "txid must be hexadecimal");
452  uint256 txid;
453  txid.SetHex(txidHex);
454 
455  int nOut = find_value(prevOut, "vout").get_int();
456  if (nOut < 0)
457  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "vout must be positive");
458 
459  string pkHex = find_value(prevOut, "scriptPubKey").get_str();
460  if (!IsHex(pkHex))
461  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "scriptPubKey must be hexadecimal");
462  vector<unsigned char> pkData(ParseHex(pkHex));
463  CScript scriptPubKey(pkData.begin(), pkData.end());
464 
465  COutPoint outpoint(txid, nOut);
466  if (mapPrevOut.count(outpoint))
467  {
468  // Complain if scriptPubKey doesn't match
469  if (mapPrevOut[outpoint] != scriptPubKey)
470  {
471  string err("Previous output scriptPubKey mismatch:\n");
472  err = err + mapPrevOut[outpoint].ToString() + "\nvs:\n"+
473  scriptPubKey.ToString();
474  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, err);
475  }
476  }
477  else
478  mapPrevOut[outpoint] = scriptPubKey;
479 
480  // if redeemScript given and not using the local wallet (private keys
481  // given), add redeemScript to the tempKeystore so it can be signed:
482  if (fGivenKeys && scriptPubKey.IsPayToScriptHash())
483  {
484  RPCTypeCheck(prevOut, map_list_of("txid", str_type)("vout", int_type)("scriptPubKey", str_type)("redeemScript",str_type));
485  Value v = find_value(prevOut, "redeemScript");
486  if (!(v == Value::null))
487  {
488  vector<unsigned char> rsData(ParseHexV(v, "redeemScript"));
489  CScript redeemScript(rsData.begin(), rsData.end());
490  tempKeystore.AddCScript(redeemScript);
491  }
492  }
493  }
494  }
495 
496 #ifdef ENABLE_WALLET
497  const CKeyStore& keystore = ((fGivenKeys || !pwalletMain) ? tempKeystore : *pwalletMain);
498 #else
499  const CKeyStore& keystore = tempKeystore;
500 #endif
501 
502  int nHashType = SIGHASH_ALL;
503  if (params.size() > 3 && params[3].type() != null_type)
504  {
505  static map<string, int> mapSigHashValues =
506  boost::assign::map_list_of
507  (string("ALL"), int(SIGHASH_ALL))
508  (string("ALL|ANYONECANPAY"), int(SIGHASH_ALL|SIGHASH_ANYONECANPAY))
509  (string("NONE"), int(SIGHASH_NONE))
510  (string("NONE|ANYONECANPAY"), int(SIGHASH_NONE|SIGHASH_ANYONECANPAY))
511  (string("SINGLE"), int(SIGHASH_SINGLE))
512  (string("SINGLE|ANYONECANPAY"), int(SIGHASH_SINGLE|SIGHASH_ANYONECANPAY))
513  ;
514  string strHashType = params[3].get_str();
515  if (mapSigHashValues.count(strHashType))
516  nHashType = mapSigHashValues[strHashType];
517  else
518  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid sighash param");
519  }
520 
521  bool fHashSingle = ((nHashType & ~SIGHASH_ANYONECANPAY) == SIGHASH_SINGLE);
522 
523  // Sign what we can:
524  for (unsigned int i = 0; i < mergedTx.vin.size(); i++)
525  {
526  CTxIn& txin = mergedTx.vin[i];
527  if (mapPrevOut.count(txin.prevout) == 0)
528  {
529  fComplete = false;
530  continue;
531  }
532  const CScript& prevPubKey = mapPrevOut[txin.prevout];
533 
534  txin.scriptSig.clear();
535  // Only sign SIGHASH_SINGLE if there's a corresponding output:
536  if (!fHashSingle || (i < mergedTx.vout.size()))
537  SignSignature(keystore, prevPubKey, mergedTx, i, nHashType);
538 
539  // ... and merge in other signatures:
540  BOOST_FOREACH(const CTransaction& txv, txVariants)
541  {
542  txin.scriptSig = CombineSignatures(prevPubKey, mergedTx, i, txin.scriptSig, txv.vin[i].scriptSig);
543  }
544  if (!VerifyScript(txin.scriptSig, prevPubKey, mergedTx, i, STANDARD_SCRIPT_VERIFY_FLAGS, 0))
545  fComplete = false;
546  }
547 
548  Object result;
549  CDataStream ssTx(SER_NETWORK, PROTOCOL_VERSION);
550  ssTx << mergedTx;
551  result.push_back(Pair("hex", HexStr(ssTx.begin(), ssTx.end())));
552  result.push_back(Pair("complete", fComplete));
553 
554  return result;
555 }

560  throw runtime_error(

570 
571  // deserialize binary data stream

577  }
