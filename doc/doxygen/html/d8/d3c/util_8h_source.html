342 template<typename T>
343 std::string HexStr(const T itbegin, const T itend, bool fSpaces=false)
344 {
345  std::string rv;
346  static const char hexmap[16] = { '0', '1', '2', '3', '4', '5', '6', '7',
347  '8', '9', 'a', 'b', 'c', 'd', 'e', 'f' };
348  rv.reserve((itend-itbegin)*3);
349  for(T it = itbegin; it < itend; ++it)
350  {
351  unsigned char val = (unsigned char)(*it);
352  if(fSpaces && it != itbegin)
353  rv.push_back(' ');
354  rv.push_back(hexmap[val>>4]);
355  rv.push_back(hexmap[val&15]);
356  }
357 
358  return rv;
359 }

490 bool TimingResistantEqual(const T& a, const T& b)
491 {
492  if (b.size() == 0) return a.size() == 0;
493  size_t accumulator = a.size() ^ b.size();
494  for (size_t i = 0; i < a.size(); i++)
495  accumulator |= a[i] ^ b[i%b.size()];
496  return accumulator == 0;
