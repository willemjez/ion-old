25 /*
26  Masternode - Proof of Service
27 
28  -- What it checks
29 
30  1.) Making sure Masternodes have their ports open
31  2.) Are responding to requests made by the network
32 
33  -- How it works
34 
35  When a block comes in, DoMasternodePOS is executed if the client is a
36  masternode. Using the deterministic ranking algorithm up to 1% of the masternode
37  network is checked each block.
38 
39  A port is opened from Masternode A to Masternode B, if successful then nothing happens.
40  If there is an error, a CMasternodeScanningError object is propagated with an error code.
41  Errors are applied to the Masternodes and a score is incremented within the masternode object,
42  after a threshold is met, the masternode goes into an error state. Each cycle the score is
43  decreased, so if the masternode comes back online it will return to the list.
44 
45  Masternodes in a error state do not receive payment.
46 
47  -- Future expansion
48 
49  We want to be able to prove the nodes have many qualities such as a specific CPU speed, bandwidth,
50  and dedicated storage. E.g. We could require a full node be a computer running 2GHz with 10GB of space.
51 
52 */

54 void ProcessMessageMasternodePOS(CNode* pfrom, std::string& strCommand, CDataStream& vRecv)
55 {
56  if(!IsSporkActive(SPORK_7_MASTERNODE_SCANNING)) return;
57  if(IsInitialBlockDownload()) return;
58 
59  if (strCommand == "mnse") //Masternode Scanning Error
60  {
61 
62  CDataStream vMsg(vRecv);
63  CMasternodeScanningError mnse;
64  vRecv >> mnse;
65 
66  CInv inv(MSG_MASTERNODE_SCANNING_ERROR, mnse.GetHash());
67  pfrom->AddInventoryKnown(inv);
68 
69  if(mapMasternodeScanningErrors.count(mnse.GetHash())){
70  return;
71  }
72  mapMasternodeScanningErrors.insert(make_pair(mnse.GetHash(), mnse));
73 
74  if(!mnse.IsValid())
75  {
76  LogPrintf("MasternodePOS::mnse - Invalid object\n");
77  return;
78  }
79 
80  CMasternode* pmnA = mnodeman.Find(mnse.vinMasternodeA);
81  if(pmnA == NULL) return;
82  if(pmnA->protocolVersion < MIN_MASTERNODE_POS_PROTO_VERSION) return;
83 
84  int nBlockHeight = pindexBest->nHeight;
85  if(nBlockHeight - mnse.nBlockHeight > 10){
86  LogPrintf("MasternodePOS::mnse - Too old\n");
87  return;
88  }
89 
90  // Lowest masternodes in rank check the highest each block
91  int a = mnodeman.GetMasternodeRank(mnse.vinMasternodeA, mnse.nBlockHeight, MIN_MASTERNODE_POS_PROTO_VERSION);
92  if(a == -1 || a > GetCountScanningPerBlock())
93  {
94  if(a != -1) LogPrintf("MasternodePOS::mnse - MasternodeA ranking is too high\n");
95  return;
96  }
97 
98  int b = mnodeman.GetMasternodeRank(mnse.vinMasternodeB, mnse.nBlockHeight, MIN_MASTERNODE_POS_PROTO_VERSION, false);
99  if(b == -1 || b < mnodeman.CountMasternodesAboveProtocol(MIN_MASTERNODE_POS_PROTO_VERSION)-GetCountScanningPerBlock())
100  {
101  if(b != -1) LogPrintf("MasternodePOS::mnse - MasternodeB ranking is too low\n");
102  return;
103  }
104 
105  if(!mnse.SignatureValid()){
106  LogPrintf("MasternodePOS::mnse - Bad masternode message\n");
107  return;
108  }
109 
110  CMasternode* pmnB = mnodeman.Find(mnse.vinMasternodeB);
111  if(pmnB == NULL) return;
112 
113  if(fDebug) LogPrintf("ProcessMessageMasternodePOS::mnse - nHeight %d MasternodeA %s MasternodeB %s\n", mnse.nBlockHeight, pmnA->addr.ToString().c_str(), pmnB->addr.ToString().c_str());
114 
115  pmnB->ApplyScanningError(mnse);
116  mnse.Relay();
117  }
118 }

145 // Check other masternodes to make sure they're running correctly
146 void CMasternodeScanning::DoMasternodePOSChecks()
147 {
148  if(!fMasterNode) return;
149  if(!IsSporkActive(SPORK_7_MASTERNODE_SCANNING)) return;
150  if(IsInitialBlockDownload()) return;
151 
152  int nBlockHeight = pindexBest->nHeight-5;
153 
154  int a = mnodeman.GetMasternodeRank(activeMasternode.vin, nBlockHeight, MIN_MASTERNODE_POS_PROTO_VERSION);
155  if(a == -1 || a > GetCountScanningPerBlock()){
156  // we don't need to do anything this block
157  return;
158  }
159 
160  // The lowest ranking nodes (Masternode A) check the highest ranking nodes (Masternode B)
161  CMasternode* pmn = mnodeman.GetMasternodeByRank(mnodeman.CountMasternodesAboveProtocol(MIN_MASTERNODE_POS_PROTO_VERSION)-a, nBlockHeight, MIN_MASTERNODE_POS_PROTO_VERSION, false);
162  if(pmn == NULL) return;
163 
164  // -- first check : Port is open
165 
166  if(!ConnectNode((CAddress)pmn->addr, NULL, true)){
167  // we couldn't connect to the node, let's send a scanning error
168  CMasternodeScanningError mnse(activeMasternode.vin, pmn->vin, SCANNING_ERROR_NO_RESPONSE, nBlockHeight);
169  mnse.Sign();
170  mapMasternodeScanningErrors.insert(make_pair(mnse.GetHash(), mnse));
171  mnse.Relay();
172  }
173 
174  // success
175  CMasternodeScanningError mnse(activeMasternode.vin, pmn->vin, SCANNING_SUCCESS, nBlockHeight);
176  mnse.Sign();
177  mapMasternodeScanningErrors.insert(make_pair(mnse.GetHash(), mnse));
178  mnse.Relay();
179 }
