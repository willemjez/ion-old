757 QString formatDurationStr(int secs)
758 {
759  QStringList strList;
760  int days = secs / 86400;
761  int hours = (secs % 86400) / 3600;
762  int mins = (secs % 3600) / 60;
763  int seconds = secs % 60;
764 
765  if (days)
766  strList.append(QString(QObject::tr("%1 d")).arg(days));
767  if (hours)
768  strList.append(QString(QObject::tr("%1 h")).arg(hours));
769  if (mins)
770  strList.append(QString(QObject::tr("%1 m")).arg(mins));
771  if (seconds || (!days && !hours && !mins))
772  strList.append(QString(QObject::tr("%1 s")).arg(seconds));
773 
774  return strList.join(" ");
775 }

777 QString formatServicesStr(uint64_t mask)
778 {
779  QStringList strList;
780 
781  // Just scan the last 8 bits for now.
782  for (int i=0; i < 8; i++) {
783  uint64_t check = 1 << i;
784  if (mask & check)
785  {
786  switch (check)
787  {
788  case NODE_NETWORK:
789  strList.append(QObject::tr("NETWORK"));
790  break;
791  default:
792  strList.append(QString("%1[%2]").arg(QObject::tr("UNKNOWN")).arg(check));
793  }
794  }
795  }
796 
797  if (strList.size())
798  return strList.join(" & ");
799  else
800  return QObject::tr("None");
801 
802 }
