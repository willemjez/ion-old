91 base_uint<BITS>& base_uint<BITS>::operator/=(const base_uint& b)
92 {
93  base_uint<BITS> div = b; // make a copy, so we can shift.
94  base_uint<BITS> num = *this; // make a copy, so we can subtract.
95  *this = 0; // the quotient.
96  int num_bits = num.bits();
97  int div_bits = div.bits();
98  if (div_bits == 0)
99  throw uint_error("Division by zero");
100  if (div_bits > num_bits) // the result is certainly 0.
101  return *this;
102  int shift = num_bits - div_bits;
103  div <<= shift; // shift so that div and nun align.
104  while (shift >= 0) {
105  if (num >= div) {
106  num -= div;
107  pn[shift / 32] |= (1 << (shift & 31)); // set a bit of the result.
108  }
109  div >>= 1; // shift back.
110  shift--;
111  }
112  // num now contains the remainder of the division.
113  return *this;

275 uint32_t uint256::GetCompact(bool fNegative) const
276 {
277  int nSize = (bits() + 7) / 8;
278  uint32_t nCompact = 0;
279  if (nSize <= 3) {
280  nCompact = GetLow64() << 8 * (3 - nSize);
281  } else {
282  uint256 bn = *this >> 8 * (nSize - 3);
283  nCompact = bn.GetLow64();
284  }
285  // The 0x00800000 bit denotes the sign.
286  // Thus, if it is already set, divide the mantissa by 256 and increase the exponent.
287  if (nCompact & 0x00800000) {
288  nCompact >>= 8;
289  nSize++;
290  }
291  assert((nCompact & ~0x007fffff) == 0);
292  assert(nSize < 256);
293  nCompact |= nSize << 24;
294  nCompact |= (fNegative && (nCompact & 0x007fffff) ? 0x00800000 : 0);
295  return nCompact;
