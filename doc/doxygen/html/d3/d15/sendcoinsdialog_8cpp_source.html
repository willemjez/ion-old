153 void SendCoinsDialog::on_sendButton_clicked()
154 {
155  if(!model || !model->getOptionsModel())
156  return;
157  QList<SendCoinsRecipient> recipients;
158  bool valid = true;
159 
160  for(int i = 0; i < ui->entries->count(); ++i)
161  {
162  SendCoinsEntry *entry = qobject_cast<SendCoinsEntry*>(ui->entries->itemAt(i)->widget());
163  if(entry)
164  {
165  if(entry->validate())
166  {
167  recipients.append(entry->getValue());
168  }
169  else
170  {
171  valid = false;
172  }
173  }
174  }
175 
176  if(!valid || recipients.isEmpty())
177  {
178  return;
179  }
180 
181  QString strFunds = tr("using") + " <b>" + tr("anonymous funds") + "</b>";
182  QString strFee = "";
183  recipients[0].inputType = ONLY_DENOMINATED;
184 
185  if(ui->checkUseStashedsend->isChecked()) {
186  recipients[0].inputType = ONLY_DENOMINATED;
187  strFunds = tr("using") + " <b>" + tr("anonymous funds") + "</b>";
188  QString strNearestAmount(
189  IonUnits::formatWithUnit(
190  model->getOptionsModel()->getDisplayUnit(), 0.1 * COIN));
191  strFee = QString(tr(
192  "(stashedsend requires this amount to be rounded up to the nearest %1)."
193  ).arg(strNearestAmount));
194  } else {
195  recipients[0].inputType = ALL_COINS;
196  strFunds = tr("using") + " <b>" + tr("any available funds (not recommended)") + "</b>";
197  }
198 
199  if(ui->checkInstantX->isChecked()) {
200  recipients[0].useInstantX = true;
201  strFunds += " ";
202  strFunds += tr("and InstantX");
203  } else {
204  recipients[0].useInstantX = false;
205  }
206 
207  // Format confirmation message
208  QStringList formatted;
209  foreach(const SendCoinsRecipient &rcp, recipients)
210  {
211  // generate bold amount string
212  QString amount = "<b>" + IonUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), rcp.amount);
213  amount.append("</b> ").append(strFunds);
214 
215  // generate monospace address string
216  QString address = "<span style='font-family: monospace;'>" + rcp.address;
217  address.append("</span>");
218 
219  QString recipientElement;
220  recipientElement = tr("%1 to %2").arg(amount, address);
221 
222  formatted.append(recipientElement);
223  }
224 
225  fNewRecipientAllowed = false;
226 
227  // request unlock only if was locked or unlocked for mixing:
228  // this way we let users unlock by walletpassphrase or by menu
229  // and make many transactions while unlocking through this dialog
230  // will call relock
231 
232  WalletModel::EncryptionStatus encStatus = model->getEncryptionStatus();
233  if(encStatus == model->Locked || encStatus == model->UnlockedForAnonymizationOnly)
234  {
235  if (!fWalletUnlockStakingOnly) {
236  WalletModel::UnlockContext ctx(model->requestUnlock());
237 
238  if(!ctx.isValid())
239  {
240  // Unlock wallet was cancelled
241  fNewRecipientAllowed = true;
242  return;
243  }
244 
245  send(recipients, strFee, formatted);
246  return;
247  } else {
248  QMessageBox::warning(this, tr("Send Transaction Failed"), tr("Wallet is currently unlocked for Staking or unlocked for StashedSend Anonymization only! \nPlease unlock your wallet manually or wait for StashedSend mixing to complete before continuing"));
249  return;
250  }
251  }
252 
253 
254  // already unlocked or not encrypted at all
255  send(recipients, strFee, formatted);
256 }

258 void SendCoinsDialog::send(QList<SendCoinsRecipient> recipients, QString strFee, QStringList formatted)
259 {
260  // prepare transaction for getting txFee earlier
261  WalletModelTransaction currentTransaction(recipients);
262  WalletModel::SendCoinsReturn prepareStatus;
263  if (model->getOptionsModel()->getCoinControlFeatures()) // coin control enabled
264  prepareStatus = model->prepareTransaction(currentTransaction, CoinControlDialog::coinControl);
265  else
266  prepareStatus = model->prepareTransaction(currentTransaction);
267 
268  // process prepareStatus and on error generate message shown to user
269  processSendCoinsReturn(prepareStatus,
270  IonUnits::formatWithUnit(model->getOptionsModel()->getDisplayUnit(), currentTransaction.getTransactionFee()));
271 
272  if(prepareStatus.status != WalletModel::OK) {
273  fNewRecipientAllowed = true;
274  return;
275  }
276 
277  CAmount txFee = currentTransaction.getTransactionFee();
278  QString questionString = tr("Are you sure you want to send?");
279  questionString.append("<br /><br />%1");
280 
281  if(txFee > 0)
282  {
283  // append fee string if a fee is required
284  questionString.append("<hr /><span style='color:#aa0000;'>");
285  questionString.append(IonUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), txFee));
286  questionString.append("</span> ");
287  questionString.append(tr("are added as transaction fee"));
288  questionString.append(" ");
289  questionString.append(strFee);
290 
291  // append transaction size
292  questionString.append(" (" + QString::number((double)currentTransaction.getTransactionSize() / 1000) + " kB)");
293  }
294 
295  // add total amount in all subdivision units
296  questionString.append("<hr />");
297  CAmount totalAmount = currentTransaction.getTotalTransactionAmount() + txFee;
298  QStringList alternativeUnits;
299  foreach(IonUnits::Unit u, IonUnits::availableUnits())
300  {
301  if(u != model->getOptionsModel()->getDisplayUnit())
302  alternativeUnits.append(IonUnits::formatHtmlWithUnit(u, totalAmount));
303  }
304 
305  // Show total amount + all alternative units
306  questionString.append(tr("Total Amount = <b>%1</b><br />= %2")
307  .arg(IonUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), totalAmount))
308  .arg(alternativeUnits.join("<br />= ")));
309 
310  // Limit number of displayed entries
311  int messageEntries = formatted.size();
312  int displayedEntries = 0;
313  for(int i = 0; i < formatted.size(); i++){
314  if(i >= MAX_SEND_POPUP_ENTRIES){
315  formatted.removeLast();
316  i--;
317  }
318  else{
319  displayedEntries = i+1;
320  }
321  }
322  questionString.append("<hr />");
323  questionString.append(tr("<b>(%1 of %2 entries displayed)</b>").arg(displayedEntries).arg(messageEntries));
324 
325  // Display message box
326  QMessageBox::StandardButton retval = QMessageBox::question(this, tr("Confirm send coins"),
327  questionString.arg(formatted.join("<br />")),
328  QMessageBox::Yes | QMessageBox::Cancel,
329  QMessageBox::Cancel);
330 
331  if(retval != QMessageBox::Yes)
332  {
333  fNewRecipientAllowed = true;
334  return;
335  }
336 
337  // now send the prepared transaction
338  WalletModel::SendCoinsReturn sendStatus = model->sendCoins(currentTransaction, CoinControlDialog::coinControl);
339  // process sendStatus and on error generate message shown to user
340  processSendCoinsReturn(sendStatus);
341 
342  if (sendStatus.status == WalletModel::OK)
343  {
344  accept();
345  CoinControlDialog::coinControl->UnSelectAll();
346  coinControlUpdateLabels();
347  }
348  fNewRecipientAllowed = true;
349 }

534 void SendCoinsDialog::processSendCoinsReturn(const WalletModel::SendCoinsReturn &sendCoinsReturn, const QString &msgArg)
535 {
536  QPair<QString, CClientUIInterface::MessageBoxFlags> msgParams;
537  // Default to a warning message, override if error message is needed
538  msgParams.second = CClientUIInterface::MSG_WARNING;
539  // This comment is specific to SendCoinsDialog usage of WalletModel::SendCoinsReturn.
540  // WalletModel::TransactionCommitFailed is used only in WalletModel::sendCoins()
541  // all others are used only in WalletModel::prepareTransaction()
542  switch(sendCoinsReturn.status)
543  {
544  case WalletModel::InvalidAddress:
545  msgParams.first = tr("The recipient address is not valid, please recheck.");
546  break;
547  case WalletModel::InvalidAmount:
548  msgParams.first = tr("The amount to pay must be larger than 0.");
549  break;
550  case WalletModel::AmountExceedsBalance:
551  msgParams.first = tr("The amount exceeds your balance.");
552  break;
553  case WalletModel::AmountWithFeeExceedsBalance:
554  msgParams.first = tr("The total exceeds your balance when the %1 transaction fee is included.").arg(msgArg);
555  break;
556  case WalletModel::DuplicateAddress:
557  msgParams.first = tr("Duplicate address found, can only send to each address once per send operation.");
558  break;
559  case WalletModel::TransactionCreationFailed:
560  msgParams.first = tr("Error: Transaction creation failed!");
561  msgParams.second = CClientUIInterface::MSG_ERROR;
562  break;
563  case WalletModel::IXTransactionCreationFailed:
564  msgParams.first = tr("InstantX doesn't support sending values that high yet. Transactions are currently limited to 1000 ION.");
565  msgParams.second = CClientUIInterface::MSG_ERROR;
566  break;
567  case WalletModel::TransactionCommitFailed:
568  msgParams.first = tr("Error: The transaction was rejected. This might happen if some of the coins in your wallet were already spent, such as if you used a copy of wallet.dat and coins were spent in the copy but not marked as spent here.");
569  msgParams.second = CClientUIInterface::MSG_ERROR;
570  break;
571  case WalletModel::AnonymizeOnlyUnlocked:
572  QMessageBox::warning(this, tr("Send Coins"),
573  tr("Error: The wallet was unlocked only to anonymize coins."),
574  QMessageBox::Ok, QMessageBox::Ok);
575  break;
576  case WalletModel::NarrationTooLong:
577  msgParams.first = tr("Error: Narration is too long.");
578  break;
579  case WalletModel::InsaneFee:
580  msgParams.first = tr("A fee %1 times higher than %2 per recipient is considered an insanely high fee.").arg(10000).arg(IonUnits::formatWithUnit(model->getOptionsModel()->getDisplayUnit(), MIN_RELAY_TX_FEE));
581  break;
582  case WalletModel::PrepareTransactionFailed:
583  return;
584  case WalletModel::OK:
585  default:
586  return;
587  }
588  emit message(tr("Send Coins"), msgParams.first, false, msgParams.second);
589 }
