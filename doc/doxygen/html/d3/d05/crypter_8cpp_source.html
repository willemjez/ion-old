134 bool EncryptAES256(const SecureString& sKey, const SecureString& sPlaintext, const std::string& sIV, std::string& sCiphertext)
135 {
136  // max ciphertext len for a n bytes of plaintext is
137  // n + AES_BLOCK_SIZE - 1 bytes
138  int nLen = sPlaintext.size();
139  int nCLen = nLen + AES_BLOCK_SIZE;
140  int nFLen = 0;
141 
142  // Verify key sizes
143  if(sKey.size() != 32 || sIV.size() != AES_BLOCK_SIZE) {
144  LogPrintf("crypter EncryptAES256 - Invalid key or block size: Key: %d sIV:%d\n", sKey.size(), sIV.size());
145  return false;
146  }
147 
148  // Prepare output buffer
149  sCiphertext.resize(nCLen);
150 
151  // Perform the encryption
152  EVP_CIPHER_CTX ctx;
153 
154  bool fOk = true;
155 
156  EVP_CIPHER_CTX_init(&ctx);
157  if (fOk) fOk = EVP_EncryptInit_ex(&ctx, EVP_aes_256_cbc(), NULL, (const unsigned char*) &sKey[0], (const unsigned char*) &sIV[0]);
158  if (fOk) fOk = EVP_EncryptUpdate(&ctx, (unsigned char*) &sCiphertext[0], &nCLen, (const unsigned char*) &sPlaintext[0], nLen);
159  if (fOk) fOk = EVP_EncryptFinal_ex(&ctx, (unsigned char*) (&sCiphertext[0])+nCLen, &nFLen);
160  EVP_CIPHER_CTX_cleanup(&ctx);
161 
162  if (!fOk) return false;
163 
164  sCiphertext.resize(nCLen + nFLen);

225 bool CCryptoKeyStore::Unlock(const CKeyingMaterial& vMasterKeyIn)
226 {
227  {
228  LOCK(cs_KeyStore);
229  if (!SetCrypted())
230  return false;
231 
232  CryptedKeyMap::const_iterator mi = mapCryptedKeys.begin();
233  for (; mi != mapCryptedKeys.end(); ++mi)
234  {
235  const CPubKey &vchPubKey = (*mi).second.first;
236  const std::vector<unsigned char> &vchCryptedSecret = (*mi).second.second;
237  CKeyingMaterial vchSecret;
238  if(!DecryptSecret(vMasterKeyIn, vchCryptedSecret, vchPubKey.GetHash(), vchSecret))
239  return false;
240  if (vchSecret.size() != 32)
241  return false;
242  CKey key;
243  key.Set(vchSecret.begin(), vchSecret.end(), vchPubKey.IsCompressed());
244  if (key.GetPubKey() == vchPubKey)
245  break;
246  return false;
247  }
248  vMasterKey = vMasterKeyIn;
249  }
250  NotifyStatusChanged(this);
251  return true;
252 }
