159  if (pwalletMain && pwalletMain->mapAddressBook.count(dest))
160  ret.push_back(Pair("account", pwalletMain->mapAddressBook[dest]));
161 #endif
162  }

206 Value verifymessage(const Array& params, bool fHelp)
207 {
208  if (fHelp || params.size() != 3)
209  throw runtime_error(
210  "verifymessage <ionaddress> <signature> <message>\n"
211  "Verify a signed message");
212 
213  string strAddress = params[0].get_str();
214  string strSign = params[1].get_str();
215  string strMessage = params[2].get_str();
216 
217  CIonAddress addr(strAddress);
218  if (!addr.IsValid())
219  throw JSONRPCError(RPC_TYPE_ERROR, "Invalid address");
220 
221  CKeyID keyID;
222  if (!addr.GetKeyID(keyID))
223  throw JSONRPCError(RPC_TYPE_ERROR, "Address does not refer to key");
224 
225  bool fInvalid = false;
226  vector<unsigned char> vchSig = DecodeBase64(strSign.c_str(), &fInvalid);
227 
228  if (fInvalid)
229  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Malformed base64 encoding");
230 
231  CHashWriter ss(SER_GETHASH, 0);
232  ss << strMessageMagic;
233  ss << strMessage;
234 
235  CPubKey pubkey;
236  if (!pubkey.RecoverCompact(ss.GetHash(), vchSig))
237  return false;
238 
239  return (pubkey.GetID() == keyID);
240 }
