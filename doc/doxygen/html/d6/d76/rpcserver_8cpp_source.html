146 string CRPCTable::help(string strCommand) const
147 {
148  string strRet;
149  set<rpcfn_type> setDone;
150  for (map<string, const CRPCCommand*>::const_iterator mi = mapCommands.begin(); mi != mapCommands.end(); ++mi)
151  {
152  const CRPCCommand *pcmd = mi->second;
153  string strMethod = mi->first;
154  // We already filter duplicates, but these deprecated screw up the sort order
155  if (strMethod.find("label") != string::npos)
156  continue;
157  if (strCommand != "" && strMethod != strCommand)
158  continue;
159 #ifdef ENABLE_WALLET
160  if (pcmd->reqWallet && !pwalletMain)
161  continue;
162 #endif
163 
164  try
165  {
166  Array params;
167  rpcfn_type pfn = pcmd->actor;
168  if (setDone.insert(pfn).second)
169  (*pfn)(params, true);
170  }
171  catch (std::exception& e)
172  {
173  // Help text is returned in an exception
174  string strHelp = string(e.what());
175  if (strCommand == "")
176  if (strHelp.find('\n') != string::npos)
177  strHelp = strHelp.substr(0, strHelp.find('\n'));
178  strRet += strHelp + "\n";
179  }
180  }
181  if (strRet == "")
182  strRet = strprintf("help: unknown command: %s\n", strCommand);
183  strRet = strRet.substr(0,strRet.size()-1);
184  return strRet;
185 }

365 bool ClientAllowed(const boost::asio::ip::address& address)
366 {
367  // Make sure that IPv4-compatible and IPv4-mapped IPv6 addresses are treated as IPv4 addresses
368  if (address.is_v6()
369  && (address.to_v6().is_v4_compatible()
370  || address.to_v6().is_v4_mapped()))
371  return ClientAllowed(address.to_v6().to_v4());
372 
373  if (address == asio::ip::address_v4::loopback()
374  || address == asio::ip::address_v6::loopback()
375  || (address.is_v4()
376  // Check whether IPv4 addresses match 127.0.0.0/8 (loopback subnet)
377  && (address.to_v4().to_ulong() & 0xff000000) == 0x7f000000))
378  return true;
379 
380  const string strAddress = address.to_string();
381  const vector<string>& vAllow = mapMultiArgs["-rpcallowip"];
382  BOOST_FOREACH(string strAllow, vAllow)
383  if (WildcardMatch(strAddress, strAllow))
384  return true;
385  return false;
386 }

471 template <typename Protocol, typename SocketAcceptorService>
472 static void RPCAcceptHandler(boost::shared_ptr< basic_socket_acceptor<Protocol, SocketAcceptorService> > acceptor,
473  ssl::context& context,
474  const bool fUseSSL,
475  AcceptedConnection* conn,
476  const boost::system::error_code& error)
477 {
478  // Immediately start accepting new connections, except when we're cancelled or our socket is closed.
479  if (error != asio::error::operation_aborted && acceptor->is_open())
480  RPCListen(acceptor, context, fUseSSL);
481 
482  AcceptedConnectionImpl<ip::tcp>* tcp_conn = dynamic_cast< AcceptedConnectionImpl<ip::tcp>* >(conn);
483 
484  // TODO: Actually handle errors
485  if (error)
486  {
487  delete conn;
488  }
489 
490  // Restrict callers by IP. It is important to
491  // do this before starting client thread, to filter out
492  // certain DoS and misbehaving clients.
493  else if (tcp_conn && !ClientAllowed(tcp_conn->peer.address()))
494  {
495  // Only send a 403 if we're not using SSL to prevent a DoS during the SSL handshake.
496  if (!fUseSSL)
497  conn->stream() << HTTPReply(HTTP_FORBIDDEN, "", false) << std::flush;
498  delete conn;
499  }
500  else {
501  ServiceConnection(conn);
502  conn->close();
503  delete conn;
504  }

733 void ServiceConnection(AcceptedConnection *conn)
734 {
735  bool fRun = true;
736  while (fRun)
737  {
738  int nProto = 0;
739  map<string, string> mapHeaders;
740  string strRequest, strMethod, strURI;
741 
742  // Read HTTP request line
743  if (!ReadHTTPRequestLine(conn->stream(), nProto, strMethod, strURI))
744  break;
745 
746  // Read HTTP message headers and body
747  ReadHTTPMessage(conn->stream(), mapHeaders, strRequest, nProto, MAX_SIZE);
748 
749  if (strURI != "/") {
750  conn->stream() << HTTPReply(HTTP_NOT_FOUND, "", false) << std::flush;
751  break;
752  }
753 
754  // Check authorization
755  if (mapHeaders.count("authorization") == 0)
756  {
757  conn->stream() << HTTPReply(HTTP_UNAUTHORIZED, "", false) << std::flush;
758  break;
759  }
760  if (!HTTPAuthorized(mapHeaders))
761  {
762  LogPrintf("ThreadRPCServer incorrect password attempt from %s\n", conn->peer_address_to_string());
763  /* Deter brute-forcing short passwords.
764  If this results in a DoS the user really
765  shouldn't have their RPC port exposed. */
766  if (mapArgs["-rpcpassword"].size() < 20)
767  MilliSleep(250);
768 
769  conn->stream() << HTTPReply(HTTP_UNAUTHORIZED, "", false) << std::flush;
770  break;
771  }
772  if (mapHeaders["connection"] == "close")
773  fRun = false;
774 
775  JSONRequest jreq;
776  try
777  {
778  // Parse request
779  Value valRequest;
780  if (!read_string(strRequest, valRequest))
781  throw JSONRPCError(RPC_PARSE_ERROR, "Parse error");
782 
783  string strReply;
784 
785  // singleton request
786  if (valRequest.type() == obj_type) {
787  jreq.parse(valRequest);
788 
789  Value result = tableRPC.execute(jreq.strMethod, jreq.params);
790 
791  // Send reply
792  strReply = JSONRPCReply(result, Value::null, jreq.id);
793 
794  // array of requests
795  } else if (valRequest.type() == array_type)
796  strReply = JSONRPCExecBatch(valRequest.get_array());
797  else
798  throw JSONRPCError(RPC_PARSE_ERROR, "Top-level object parse error");
799 
800  conn->stream() << HTTPReply(HTTP_OK, strReply, fRun) << std::flush;
801  }
802  catch (Object& objError)
803  {
804  ErrorReply(conn->stream(), objError, jreq.id);
805  break;
806  }
807  catch (std::exception& e)
808  {
809  ErrorReply(conn->stream(), JSONRPCError(RPC_PARSE_ERROR, e.what()), jreq.id);
810  break;
811  }
812  }
813 }

816 {
