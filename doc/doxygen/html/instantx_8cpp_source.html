403 bool CheckForConflictingLocks(CTransaction& tx)
404 {
405  /*
406  It's possible (very unlikely though) to get 2 conflicting transaction locks approved by the network.
407  In that case, they will cancel each other out.
408 
409  Blocks could have been rejected during this time, which is OK. After they cancel out, the client will
410  rescan the blocks and find they're acceptable and then take the chain with the most work.
411  */
412  BOOST_FOREACH(const CTxIn& in, tx.vin){
413  if(mapLockedInputs.count(in.prevout)){
414  if(mapLockedInputs[in.prevout] != tx.GetHash()){
415  LogPrintf("InstantX::CheckForConflictingLocks - found two complete conflicting locks - removing both. %s %s", tx.GetHash().ToString().c_str(), mapLockedInputs[in.prevout].ToString().c_str());
416  if(mapTxLocks.count(tx.GetHash())) mapTxLocks[tx.GetHash()].nExpiration = GetTime();
417  if(mapTxLocks.count(mapLockedInputs[in.prevout])) mapTxLocks[mapLockedInputs[in.prevout]].nExpiration = GetTime();
418  return true;
419  }
420  }
421  }
422 
423  return false;
