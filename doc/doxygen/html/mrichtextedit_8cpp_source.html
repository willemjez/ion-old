201 void MRichTextEdit::textLink(bool checked) {
202  bool unlink = false;
203  QTextCharFormat fmt;
204  if (checked) {
205  QString url = f_textedit->currentCharFormat().anchorHref();
206  bool ok;
207  QString newUrl = QInputDialog::getText(this, tr("Create a link"),
208  tr("Link URL:"), QLineEdit::Normal,
209  url,
210  &ok);
211  if (ok) {
212  fmt.setAnchor(true);
213  fmt.setAnchorHref(newUrl);
214  fmt.setForeground(QApplication::palette().color(QPalette::Link));
215  fmt.setFontUnderline(true);
216  } else {
217  unlink = true;
218  }
219  } else {
220  unlink = true;
221  }
222  if (unlink) {
223  fmt.setAnchor(false);
224  fmt.setForeground(QApplication::palette().color(QPalette::Text));
225  fmt.setFontUnderline(false);
226  }
227  mergeFormatOnWordOrSelection(fmt);
228 }

230 void MRichTextEdit::textStyle(int index) {
231  QTextCursor cursor = f_textedit->textCursor();
232  cursor.beginEditBlock();
233 
234  // standard
235  if (!cursor.hasSelection()) {
236  cursor.select(QTextCursor::BlockUnderCursor);
237  }
238  QTextCharFormat fmt;
239  cursor.setCharFormat(fmt);
240  f_textedit->setCurrentCharFormat(fmt);
241 
242  if (index == ParagraphHeading1
243  || index == ParagraphHeading2
244  || index == ParagraphHeading3
245  || index == ParagraphHeading4 ) {
246  if (index == ParagraphHeading1) {
247  fmt.setFontPointSize(m_fontsize_h1);
248  }
249  if (index == ParagraphHeading2) {
250  fmt.setFontPointSize(m_fontsize_h2);
251  }
252  if (index == ParagraphHeading3) {
253  fmt.setFontPointSize(m_fontsize_h3);
254  }
255  if (index == ParagraphHeading4) {
256  fmt.setFontPointSize(m_fontsize_h4);
257  }
258  if (index == ParagraphHeading2 || index == ParagraphHeading4) {
259  fmt.setFontItalic(true);
260  }
261 
262  fmt.setFontWeight(QFont::Bold);
263  }
264  if (index == ParagraphMonospace) {
265  fmt = cursor.charFormat();
266  fmt.setFontFamily("Monospace");
267  fmt.setFontStyleHint(QFont::Monospace);
268  fmt.setFontFixedPitch(true);
269  }
270  cursor.setCharFormat(fmt);
271  f_textedit->setCurrentCharFormat(fmt);
272 
273  cursor.endEditBlock();
274 }

335 void MRichTextEdit::slotCursorPositionChanged() {
336  QTextList *l = f_textedit->textCursor().currentList();
337  if (m_lastBlockList && (l == m_lastBlockList || (l != 0 && m_lastBlockList != 0
338  && l->format().style() == m_lastBlockList->format().style()))) {
339  return;
340  }
341  m_lastBlockList = l;
342  if (l) {
343  QTextListFormat lfmt = l->format();
344  if (lfmt.style() == QTextListFormat::ListDisc) {
345  f_list_bullet->setChecked(true);
346  f_list_ordered->setChecked(false);
347  } else if (lfmt.style() == QTextListFormat::ListDecimal) {
348  f_list_bullet->setChecked(false);
349  f_list_ordered->setChecked(true);
350  } else {
351  f_list_bullet->setChecked(false);
352  f_list_ordered->setChecked(false);
353  }
354  } else {
355  f_list_bullet->setChecked(false);
356  f_list_ordered->setChecked(false);
357  }

360 void MRichTextEdit::fontChanged(const QFont &f) {
361  f_fontsize->setCurrentIndex(f_fontsize->findText(QString::number(f.pointSize())));
362  f_bold->setChecked(f.bold());
363  f_italic->setChecked(f.italic());
364  f_underline->setChecked(f.underline());
365  f_strikeout->setChecked(f.strikeOut());
366  if (f.pointSize() == m_fontsize_h1) {
367  f_paragraph->setCurrentIndex(ParagraphHeading1);
368  } else if (f.pointSize() == m_fontsize_h2) {
369  f_paragraph->setCurrentIndex(ParagraphHeading2);
370  } else if (f.pointSize() == m_fontsize_h3) {
371  f_paragraph->setCurrentIndex(ParagraphHeading3);
372  } else if (f.pointSize() == m_fontsize_h4) {
373  f_paragraph->setCurrentIndex(ParagraphHeading4);
374  } else {
375  if (f.fixedPitch() && f.family() == "Monospace") {
376  f_paragraph->setCurrentIndex(ParagraphMonospace);
377  } else {
378  f_paragraph->setCurrentIndex(ParagraphStandard);
379  }
380  }
381  if (f_textedit->textCursor().currentList()) {
382  QTextListFormat lfmt = f_textedit->textCursor().currentList()->format();
383  if (lfmt.style() == QTextListFormat::ListDisc) {
384  f_list_bullet->setChecked(true);
385  f_list_ordered->setChecked(false);
386  } else if (lfmt.style() == QTextListFormat::ListDecimal) {
387  f_list_bullet->setChecked(false);
388  f_list_ordered->setChecked(true);
389  } else {
390  f_list_bullet->setChecked(false);
391  f_list_ordered->setChecked(false);
392  }
393  } else {
394  f_list_bullet->setChecked(false);
395  f_list_ordered->setChecked(false);

422 QString MRichTextEdit::toHtml() const {
423  QString s = f_textedit->toHtml();
424  // convert emails to links
425  s = s.replace(QRegExp("(<[^a][^>]+>(?:<span[^>]+>)?|\\s)([a-zA-Z\\d]+@[a-zA-Z\\d]+\\.[a-zA-Z]+)"), "\\1<a href=\"mailto:\\2\">\\2</a>");
426  // convert links
427  s = s.replace(QRegExp("(<[^a][^>]+>(?:<span[^>]+>)?|\\s)((?:https?|ftp|file)://[^\\s'\"<>]+)"), "\\1<a href=\"\\2\">\\2</a>");
428  // see also: Utils::linkify()
429  return s;
430 }
